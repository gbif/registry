1   package org.gbif.registry.search;
2   
3   import java.util.regex.Pattern;
4   
5   import org.xml.sax.Attributes;
6   import org.xml.sax.SAXException;
7   import org.xml.sax.helpers.DefaultHandler;
8   
9   /**
10   * A simple metadata sax base handler that collects all character data inside elements into a string buffer, resetting
11   * the buffer with every element start and storing the string version of the buffer in this.content when the end of the
12   * element is reached.
13   */
14  public class FullTextSaxHandler extends DefaultHandler {
15  
16    private StringBuffer content;
17    private Pattern normWhitespace = Pattern.compile("[\\s\n\r\t]+");
18  
19    @Override
20    public void startDocument() throws SAXException {
21      super.startDocument();
22      content = new StringBuffer(" ");
23    }
24  
25    @Override
26    public void characters(char[] ch, int start, int length) {
27      content.append(ch, start, length);
28    }
29  
30    @Override
31    public void startElement(String uri, String localName, String qName, Attributes attributes) throws SAXException {
32      // as we remove tags we add whitespace to delimit words
33      content.append(" ");
34      // add attribute content to the full text
35      for (int idx = attributes.getLength(); idx > 0; idx--) {
36        content.append(attributes.getValue(idx - 1));
37        content.append(" ");
38      }
39    }
40  
41    public String getFullText() {
42      return normWhitespace.matcher(content.toString()).replaceAll(" ").trim();
43    }
44  }
