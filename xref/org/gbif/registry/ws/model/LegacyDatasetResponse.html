16   * </br>
17   * JAXB annotations allow the class to be converted into an XML document or JSON response. @XmlElement is used to
18   * specify element names that consumers of legacy services expect to find.
19   */

37    public LegacyDatasetResponse(Dataset dataset, Contact contact) {
38      key = dataset.getKey() == null ? "" : dataset.getKey().toString();
39      organisationKey = dataset.getPublishingOrganizationKey() == null ? "" : dataset.getPublishingOrganizationKey().toString();
40      name = dataset.getTitle() == null ? "" : dataset.getTitle();
41      description = dataset.getDescription() == null ? "" : dataset.getDescription();
42      descriptionLanguage = dataset.getLanguage() == null ? "" : dataset.getLanguage().getIso2LetterCode();
43      nameLanguage = dataset.getLanguage() == null ? "" : dataset.getLanguage().getIso2LetterCode();
44      homepageURL = dataset.getHomepage() == null ? "" : dataset.getHomepage().toString();
45      primaryContactAddress = contact == null || contact.getAddress().isEmpty() ? "" :
46        Strings.nullToEmpty(contact.getAddress().get(0));
47      primaryContactDescription = contact == null ? "" : Strings.nullToEmpty(contact.getDescription());
48      primaryContactEmail = contact == null || contact.getEmail().isEmpty() ? "" :
49        Strings.nullToEmpty(contact.getEmail().get(0));
50      primaryContactPhone = contact == null || contact.getPhone().isEmpty() ? "" :
51        Strings.nullToEmpty(contact.getPhone().get(0));
52      primaryContactName = contact == null
53        ? ""
54        : Strings.nullToEmpty(contact.getFirstName()) + " " + Strings.nullToEmpty(contact.getLastName());
55      // conversion of contact type, defaulting to ""
56      primaryContactType = "";
57      if (contact != null) {
58        ContactType type = contact.getType();
59        if (type != null) {
60          if (type.compareTo(ContactType.ADMINISTRATIVE_POINT_OF_CONTACT) == 0) {
61            primaryContactType = LegacyResourceConstants.ADMINISTRATIVE_CONTACT_TYPE;
62          } else if (type.compareTo(ContactType.TECHNICAL_POINT_OF_CONTACT) == 0) {
63            primaryContactType = LegacyResourceConstants.TECHNICAL_CONTACT_TYPE;
64          }
65        }
66      }
67    }
