1   package org.gbif.registry.metasync.protocols.tapir.model.search;
2   
3   import com.google.common.base.Objects;
4   import org.apache.commons.digester3.annotations.rules.ObjectCreate;
5   import org.apache.commons.digester3.annotations.rules.SetProperty;
6   
7   /**
8    * Class used to represent the response from a TAPIR search request, sent in order to discover the number of records.
9    */
10  @ObjectCreate(pattern = "response/search/summary")
11  public class TapirSearch {
12  
13    @SetProperty(pattern = "response/search/summary", attributeName = "totalMatched")
14    private int numberOfRecords;
15  
16    /**
17     * Get the total number of records for the Dataset behind the endpoint.
18     *
19     * @return total number of records for the Dataset
20     */
21    public int getNumberOfRecords() {
22      return numberOfRecords;
23    }
24  
25    public void setNumberOfRecords(int numberOfRecords) {
26      this.numberOfRecords = numberOfRecords;
27    }
28  
29    @Override
30    public String toString() {
31      return Objects.toStringHelper(this)
32        .toString();
33    }
34  }
