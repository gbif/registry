100   @POST
101   @Trim
102   @Transactional
103   @Validate(groups = {PrePersist.class, Default.class})
104   @RolesAllowed(ADMIN_ROLE)
105   @Override
106   public void create(@Valid @NotNull @Trim Download occurrenceDownload) {
107     occurrenceDownload.setDoi(doiGenerator.newDownloadDOI());
108     occurrenceDownloadMapper.create(occurrenceDownload);
109   }

111   @GET
112   @Path("{key}")
113   @Nullable
114   @NullToNotFound
115   @Override
116   public Download get(@PathParam("key") String key) {
117     Download download = occurrenceDownloadMapper.get(key);
118     if (download != null) { // the user can request a non-existing download
119       clearSensitiveData(securityContext, download);
120     }
121     return download;
122   }

124   /**
125    * Lists all the downloads. This operation can be executed by role ADMIN only.
126    */
127   @GET
128   @RolesAllowed(ADMIN_ROLE)
129   @Override
130   public PagingResponse<Download> list(@Context Pageable page, @Nullable @QueryParam("status") Set<Download.Status> status) {
131     if(status == null ||status.isEmpty()) {
132       return new PagingResponse<Download>(page, (long) occurrenceDownloadMapper.count(), occurrenceDownloadMapper.list(page));
133     } else {
134       return new PagingResponse<Download>(page, (long) occurrenceDownloadMapper.countByStatus(status), occurrenceDownloadMapper.listByStatus(page,status));
135     }
136   }

150   @PUT
151   @Path("{key}")
152   @Transactional
153   @Override
154   public void update(Download download) {
155     // The current download is retrieved because its user could be modified during the update
156     Download currentDownload = get(download.getKey());
157     Preconditions.checkNotNull(currentDownload);
158     checkUserIsInSecurityContext(currentDownload.getRequest().getCreator(), securityContext);
159     User user = userService.get(securityContext.getUserPrincipal().getName());
160     updateDownloadDOI(download, currentDownload, user);
161     occurrenceDownloadMapper.update(download);
162   }

164   /**
165    * Updates the download DOI according to the download status.
166    * If the download succeeded its DOI is registered; if the download status is one the FAILED_STATES
167    * the DOI is removed, otherwise doesn't nothing.
168    */
169   private void updateDownloadDOI(Download download, Download previousDownload, User user){
170     if(download.isAvailable() && previousDownload.getStatus() != Download.Status.SUCCEEDED){
171       try {
172         doiGenerator.registerDownload(download.getDoi(), buildMetadata(download, user), download.getKey());
173       } catch(Exception error) {
174         LOG.error(DOI_SMTP, "Invalid metadata for download {} with doi {} ", download.getKey(), download.getDoi(),  error);
175       }
176 
177     } else if(FAILED_STATES.contains(download.getStatus())){
178       doiGenerator.delete(download.getDoi());
179     }
180   }

182   /**
183    * Creates the DataCite metadata for a download object.
184    */
185   private DataCiteMetadata buildMetadata(Download download, User user) {
186     List<DatasetOccurrenceDownloadUsage> response = null;
187     List<DatasetOccurrenceDownloadUsage> usages = Lists.newArrayList();
188     PagingRequest pagingRequest = new PagingRequest(0, USAGES_PAGE_SIZE);
189 
190     while (response == null || !response.isEmpty()) {
191       response = datasetOccurrenceDownloadMapper.listByDownload(download.getKey(), pagingRequest);
192       usages.addAll(response);
193       pagingRequest.nextPage();
194     }
195 
196     return DataCiteConverter.convert(download, user, usages, titleLookup);
197   }
