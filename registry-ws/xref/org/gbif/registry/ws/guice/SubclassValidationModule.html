16  package org.gbif.registry.ws.guice;
17  
18  import javax.validation.MessageInterpolator;
19  import javax.validation.TraversableResolver;
20  import javax.validation.spi.ConfigurationState;
21  import javax.validation.spi.ValidationProvider;
22  
23  import com.google.inject.AbstractModule;
24  import com.google.inject.Scopes;
25  import com.google.inject.TypeLiteral;
26  import com.google.inject.matcher.Matchers;
27  import org.aopalliance.intercept.MethodInterceptor;
28  import org.apache.bval.guice.ConfigurationStateProvider;
29  import org.apache.bval.guice.Validate;
30  import org.apache.bval.guice.ValidateMethodInterceptor;
31  import org.apache.bval.jsr303.ApacheValidationProvider;
32  import org.apache.bval.jsr303.DefaultMessageInterpolator;
33  import org.apache.bval.jsr303.resolver.DefaultTraversableResolver;
34  
35  // TODO: Why this class?
36  public class SubclassValidationModule extends AbstractModule {
37  
38    /**
39     * {@inheritDoc}
40     */
41    @Override
42    protected void configure() {
43      // apache bval bootstrap
44      this.bind(MessageInterpolator.class).to(DefaultMessageInterpolator.class).in(Scopes.SINGLETON);
45      this.bind(TraversableResolver.class).to(DefaultTraversableResolver.class).in(Scopes.SINGLETON);
46      // this.bind(ConstraintValidatorFactory.class).to(GuiceAwareConstraintValidatorFactory.class);
47      this.bind(new TypeLiteral<ValidationProvider<?>>() {}).to(ApacheValidationProvider.class).in(Scopes.SINGLETON);
48      this.bind(ConfigurationState.class).toProvider(ConfigurationStateProvider.class).in(Scopes.SINGLETON);
49      // this.bind(ValidatorFactory.class).toProvider(ValidatorFactoryProvider.class).in(Scopes.SINGLETON);
50      // this.bind(Validator.class).toProvider(ValidatorProvider.class);
51  
52      // AOP stuff
53      MethodInterceptor validateMethodInterceptor = new ValidateMethodInterceptor();
54      this.binder().requestInjection(validateMethodInterceptor);
55      this.bindInterceptor(Matchers.any(), Matchers.annotatedWith(Validate.class), validateMethodInterceptor);
56    }
57  
58  }
