13  package org.gbif.registry.persistence.mapper;
14  
15  import org.gbif.api.model.registry.Contact;
16  import org.gbif.api.vocabulary.ContactType;
17  
18  import java.util.List;
19  import java.util.UUID;
20  
21  import org.apache.ibatis.annotations.Param;
22  
23  public interface ContactableMapper {
24  
25    int addContact(
26      @Param("targetEntityKey") UUID entityKey,
27      @Param("contactKey") int contactKey,
28      @Param("type") ContactType contactType,
29      @Param("isPrimary") boolean isPrimary
30      );
31  
32    /**
33     * Updates the current primary contact of the given type if it exists and makes it non primary.
34     */
35    void updatePrimaryContacts(@Param("targetEntityKey") UUID entityKey, @Param("type") ContactType contactType);
36  
37    /**
38     * Updates the contact if it exists, updating the type and is_primary columns.
39     */
40    void updateContact(@Param("targetEntityKey") UUID entityKey, @Param("contactKey") Integer contactKey,
41      @Param("type") ContactType contactType, @Param("primary") boolean primary);
42  
43    int deleteContact(@Param("targetEntityKey") UUID entityKey, @Param("contactKey") int contactKey);
44  
45    /**
46     * Delete all contacts associated with the given entity
47     */
48    int deleteContacts(@Param("targetEntityKey") UUID entityKey);
49  
50    List<Contact> listContacts(@Param("targetEntityKey") UUID targetEntityKey);
51  
52    /**
53     * Checks if the contact is associated with the entity.
54     * 
55     * @return true only if the contact and entity are related
56     */
57    Boolean areRelated(@Param("targetEntityKey") UUID targetEntityKey, @Param("contactKey") int contactKey);
58  }
