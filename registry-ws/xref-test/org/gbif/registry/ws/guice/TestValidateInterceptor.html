22  /**
23   * An interceptor that will set the createdBy and or modifiedBy fields on object being modified in methods having
24   * @Validate. The object must either extend NetworkEntity, or be a Contact, Endpoint, MachineTag, Tag, Identifier, or
25   * Comment.
26   * </br>
27   * This is only needed server side running the web service tests, since the tests call the interface service method,
28   * not the HTTP resource method where the createdBy and modifiedBy fields are actually set.
29   */
30  public class TestValidateInterceptor implements MethodInterceptor {
31  
32    private static final Logger LOG = LoggerFactory.getLogger(TestValidateInterceptor.class);
33  
34    /**
35     * Sets up method level interception for those methods annotated with {@link Validate}.
36     */
37    public static Module newMethodInterceptingModule() {
38      return new AbstractModule() {
39  
40        @Override
41        protected void configure() {
42          MethodInterceptor validateKyleMethodInterceptor = new TestValidateInterceptor();
43          this.binder().requestInjection(validateKyleMethodInterceptor);
44          this.bindInterceptor(Matchers.any(), Matchers.annotatedWith(Validate.class), validateKyleMethodInterceptor);
45        }
46      };
47    }
48  
49    @Override
50    public Object invoke(MethodInvocation invocation) throws Throwable {
51      Validate validate = invocation.getMethod().getAnnotation(Validate.class); // ensure it is annotated
52      if (validate != null) {
53        for (int i = 0; i < invocation.getArguments().length; i++) {
54          Object arg = invocation.getArguments()[i];
55          if (arg instanceof NetworkEntity ||
56            arg instanceof Comment ||
57            arg instanceof Contact ||
58            arg instanceof Endpoint ||
59            arg instanceof Identifier ||
60            arg instanceof MachineTag ||
61            arg instanceof Tag) {
62            addRequiredFields(invocation.getArguments()[i]);
63          }
64        }
65      }
66      return invocation.proceed();
67    }
68  
69    /**
70     * Add required fields createdBy and modifiedBy (where possible) to target object which either extends NetworkEntity
71     * or is a Contact, Endpoint, MachineTag, Tag, Identifier, or Comment.
72     *
73     * @param target object
74     */
75    private void addRequiredFields(Object target) {
76      WrapDynaBean wrapped = new WrapDynaBean(target);
77      DynaClass dynaClass = wrapped.getDynaClass();
78      // update createdBy field
79      if (dynaClass.getDynaProperty("createdBy") != null) {
80        wrapped.set("createdBy", "WS TEST");
81      }
82      // update modifiedBy field
83      if (dynaClass.getDynaProperty("modifiedBy") != null) {
84        wrapped.set("modifiedBy", "WS TEST");
85      }
86    }
87  }
