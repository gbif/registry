58    /**
59     * The test sends a get Organization (GET) request with no parameters and .json, signifying that the response must be
60     * JSON.
61     */
62    @Test
63    public void testGetOrganizationJSON() throws IOException, URISyntaxException, SAXException {
64      // persist new organization (IPT hosting organization)
65      Organization organization = Organizations.newPersistedInstance();
66      Contact c = Contacts.newInstance();
67      c.setType(ContactType.TECHNICAL_POINT_OF_CONTACT);
68      organizationService.addContact(organization.getKey(), c);
69  
70      // construct request uri
71      String uri = Requests.getRequestUri("/registry/organisation/" + organization.getKey().toString() + ".json");
72  
73      // send GET request with no credentials
74      HttpUtil.Response result = Requests.http.get(uri);
75  
76      // JSON expected
77      assertTrue(result.content.startsWith("{"));
78      JsonNode rootNode = objectMapper.readTree(result.content);
79  
80      // JSON object expected
81      assertEquals(15, rootNode.size());
82  
83      assertEquals(organization.getKey().toString(), rootNode.get("key").getTextValue());
84      assertEquals(organization.getTitle(), rootNode.get("name").getTextValue());
85      assertEquals(organization.getLanguage().getIso2LetterCode(), rootNode.get("nameLanguage").getTextValue());
86      assertEquals(organization.getDescription(), rootNode.get("description").getTextValue());
87      assertEquals(organization.getLanguage().getIso2LetterCode(), rootNode.get("descriptionLanguage").getTextValue());
88      assertEquals(organization.getHomepage().toString(), rootNode.get("homepageURL").getTextValue());
89      assertEquals(LegacyResourceConstants.TECHNICAL_CONTACT_TYPE, rootNode.get("primaryContactType").getTextValue());
90      assertEquals("Tim Robertson", rootNode.get("primaryContactName").getTextValue());
91      assertEquals("trobertson@gbif.org", rootNode.get("primaryContactEmail").getTextValue());
92      assertEquals("Universitetsparken 15", rootNode.get("primaryContactAddress").getTextValue());
93      assertEquals("+45 28261487", rootNode.get("primaryContactPhone").getTextValue());
94      assertEquals(organization.getEndorsingNodeKey().toString(), rootNode.get("nodeKey").getTextValue());
95      assertEquals("The UK National Node", rootNode.get("nodeName").getTextValue());
96      assertEquals("", rootNode.get("nodeContactEmail").getTextValue());
97    }
