074  /**
075   * The test begins by persisting a new Organization, and installation associated to that Organization.
076   * </br>
077   * Then, it sends a register Dataset (POST) request to create a new Dataset owned by this organization. The request
078   * does not contain an installation key, but it can be inferred that it should use the key of the one and only
079   * installation belonging to the publishing organization. The request also doesn't create any endpoints, that's done in
080   * a separate legacy ws call. Therefore, the DatasetType defaults to METADATA.
081   * </br>
082   * Upon receiving an HTTP Response, the test parses its XML content in order to extract the registered Dataset UUID
083   * for example. The content is parsed exactly the same way as the GBRDS WS consumer would do it.
084   * </br>
085   * Last, the test validates that the dataset was persisted correctly.
086   */
087  @Test
088  public void testRegisterLegacyDataset() throws IOException, URISyntaxException, SAXException {
089    // persist new organization (Dataset publishing organization)
090    Organization organization = Organizations.newPersistedInstance();
091    UUID organizationKey = organization.getKey();
092
093    // persist new installation of type IPT
094    Installation installation = Installations.newPersistedInstance(organizationKey);
095    UUID installationKey = installation.getKey();
096
097    // populate params for ws, without installation key
098    List<NameValuePair> data = buildLegacyDatasetParameters(organizationKey);
099    UrlEncodedFormEntity uefe = new UrlEncodedFormEntity(data, Charset.forName("UTF-8"));
100
101    // construct request uri
102    String uri = Requests.getRequestUri("/registry/resource");
103
104    // send POST request with credentials
105    HttpUtil.Response result = Requests.http.post(uri, null, null, Organizations.credentials(organization), uefe);
106
107    // parse newly registered IPT key (UUID)
108    Parsers.saxParser.parse(Parsers.getUtf8Stream(result.content), Parsers.legacyDatasetResponseHandler);
109    assertNotNull("Registered Dataset key should be in response",
110      UUID.fromString(Parsers.legacyDatasetResponseHandler.key));
111
112    // some information that should have been updated
113    Dataset dataset =
114      validatePersistedLegacyDataset(UUID.fromString(Parsers.legacyDatasetResponseHandler.key), organizationKey,
115        installationKey);
116
117    // some additional information to check
118    assertNotNull(dataset.getCreatedBy());
119    assertNotNull(dataset.getModifiedBy());
120  }

122  /**
123   * The test begins by persisting a new Organization, Installation associated to the Organization, and Dataset
124   * associated to the Organization. A primary contact and endpoint is then added to the Dataset.
125   * </br>
126   * Then, it sends an update Dataset (POST) request to update the same Dataset. The request does not have the primary
127   * contact, endpoint, or installation key form parameters. Since the organization only has 1 installation anyways,
128   * it will be inferred that the dataset belongs to this one.
129   * </br>
130   * Upon receiving an HTTP Response, the test parses its XML content in order to extract the registered Dataset UUID
131   * for example. It also ensures that the primary contact and endpoints still exist.
132   */
133  @Test
134  public void testUpdateLegacyDatasetWithNoContactNoEndpointNoInstallationKey()
135    throws IOException, URISyntaxException, SAXException {
136    // persist new organization (IPT hosting organization)
137    Organization organization = Organizations.newPersistedInstance();
138    UUID organizationKey = organization.getKey();
139
140    // persist new installation of type IPT
141    Installation installation = Installations.newPersistedInstance(organizationKey);
142    UUID installationKey = installation.getKey();
143
144    // persist new Dataset associated to installation
145    Dataset dataset = Datasets.newPersistedInstance(organizationKey, installationKey);
146    UUID datasetKey = dataset.getKey();
147    // add primary contact to Dataset
148    Contact c = Contacts.newInstance();
149    c.setType(ContactType.TECHNICAL_POINT_OF_CONTACT);
150    datasetService.addContact(datasetKey, c);
151    // add endpoint to Dataset
152    Endpoint e = Endpoints.newInstance();
153    datasetService.addEndpoint(datasetKey, e);
154
155    // validate it
156    validateExistingDataset(dataset, organizationKey, installationKey);
157
158    // before sending the update POST request, count the number of datasets, contacts and endpoints
159    assertEquals(1, datasetService.list(new PagingRequest(0, 10)).getResults().size());
160    assertEquals(1, datasetService.listEndpoints(datasetKey).size());
161    assertEquals(1, datasetService.listContacts(datasetKey).size());
162
163    // some information never going to change
164    Date created = dataset.getCreated();
165    assertNotNull(created);
166    String createdBy = dataset.getCreatedBy();
167    assertNotNull(createdBy);
168
169    // populate params for ws
170    List<NameValuePair> data = new ArrayList<NameValuePair>();
171    // main
172    data.add(new BasicNameValuePair(LegacyResourceConstants.NAME_PARAM, Requests.DATASET_NAME));
173    data.add(new BasicNameValuePair(LegacyResourceConstants.NAME_LANGUAGE_PARAM, Requests.DATASET_NAME_LANGUAGE));
174    data.add(new BasicNameValuePair(LegacyResourceConstants.DESCRIPTION_PARAM, Requests.DATASET_DESCRIPTION));
175    data.add(new BasicNameValuePair(LegacyResourceConstants.DOI_PARAM, Requests.DOI));
176    data.add(new BasicNameValuePair(LegacyResourceConstants.DESCRIPTION_LANGUAGE_PARAM,
177      Requests.DATASET_DESCRIPTION_LANGUAGE));
178    data.add(new BasicNameValuePair(LegacyResourceConstants.HOMEPAGE_URL_PARAM, Requests.DATASET_HOMEPAGE_URL));
179    data.add(new BasicNameValuePair(LegacyResourceConstants.LOGO_URL_PARAM, Requests.DATASET_LOGO_URL));
180    // add additional ipt and organisation parameters
181    data.add(new BasicNameValuePair(LegacyResourceConstants.ORGANIZATION_KEY_PARAM, organizationKey.toString()));
182
183    UrlEncodedFormEntity uefe = new UrlEncodedFormEntity(data, Charset.forName("UTF-8"));
184
185    // construct request uri
186    String uri = Requests.getRequestUri("/registry/resource/" + datasetKey.toString());
187
188    // send POST request with credentials
189    HttpUtil.Response result = Requests.http.post(uri, null, null, Organizations.credentials(organization), uefe);
190
191    // if Jersey was responding with default UTF-8 (fine in Intellij, not fine via maven) conversion below not needed
192    String st = new String(result.content.getBytes(), Charsets.UTF_8);
193    // parse updated registered Dataset key (UUID)
194
195    Parsers.saxParser.parse(Parsers.getUtf8Stream(st), Parsers.legacyDatasetResponseHandler);
196
197    assertNotNull("Updated Dataset key should be in response", Parsers.legacyDatasetResponseHandler.key);
198    assertEquals(datasetKey.toString(), Parsers.legacyDatasetResponseHandler.key);
199    assertNotNull("Updated Dataset organizationKey should be in response",
200      Parsers.legacyDatasetResponseHandler.organisationKey);
201    assertEquals(organizationKey.toString(), Parsers.legacyDatasetResponseHandler.organisationKey);
202
203    // make some additional assertions that the update was successful
204    // retrieve installation anew
205    dataset = datasetService.get(datasetKey);
206
207    assertNotNull("Dataset should be present", dataset);
208    assertEquals(organizationKey, dataset.getPublishingOrganizationKey());
209    assertEquals(installationKey, dataset.getInstallationKey());
210    assertEquals(DatasetType.OCCURRENCE, dataset.getType());
211    assertEquals(Requests.DATASET_NAME, dataset.getTitle());
212    assertEquals(Requests.DATASET_NAME_LANGUAGE, dataset.getLanguage().getIso2LetterCode());
213    assertEquals(Requests.DATASET_DESCRIPTION, dataset.getDescription());
214    assertEquals(Requests.DATASET_HOMEPAGE_URL, dataset.getHomepage().toString());
215    assertEquals(Requests.DATASET_LOGO_URL, dataset.getLogoUrl().toString());
216    assertNotNull(dataset.getCreated());
217    assertEquals(created.toString(), dataset.getCreated().toString());
218    assertEquals(createdBy, dataset.getCreatedBy());
219    assertNotNull(dataset.getModified());
220  }

554    data.add(new BasicNameValuePair(LegacyResourceConstants.HOMEPAGE_URL_PARAM, Requests.DATASET_HOMEPAGE_URL));
555    data.add(new BasicNameValuePair(LegacyResourceConstants.LOGO_URL_PARAM, Requests.DATASET_LOGO_URL));

557    // primary contact
558    data.add(new BasicNameValuePair(LegacyResourceConstants.PRIMARY_CONTACT_TYPE_PARAM,
559      Requests.DATASET_PRIMARY_CONTACT_TYPE));

562    data.add(new BasicNameValuePair(LegacyResourceConstants.PRIMARY_CONTACT_NAME_PARAM,
563      Requests.DATASET_PRIMARY_CONTACT_NAME));
564    data.add(new BasicNameValuePair(LegacyResourceConstants.PRIMARY_CONTACT_ADDRESS_PARAM,
565      Requests.DATASET_PRIMARY_CONTACT_ADDRESS.get(0)));
566    data.add(new BasicNameValuePair(LegacyResourceConstants.PRIMARY_CONTACT_PHONE_PARAM,
567      Requests.DATASET_PRIMARY_CONTACT_PHONE.get(0)));

569      Requests.DATASET_PRIMARY_CONTACT_DESCRIPTION));
570
571    // add additional ipt and organisation parameters
572    data.add(new BasicNameValuePair(LegacyResourceConstants.ORGANIZATION_KEY_PARAM, organizationKey.toString()));

576  /**
577   * Retrieve persisted Legacy (GBRDS) dataset, and make a series of assertions to ensure it has been properly
578   * persisted.
579   *
580   * @param datasetKey installation key (UUID)
581   * @param organizationKey installation publishing organization key
582   * @return validated installation
583   */
584  private Dataset validatePersistedLegacyDataset(UUID datasetKey, UUID organizationKey, UUID installationKey) {
585    // retrieve installation anew
586    Dataset dataset = datasetService.get(datasetKey);
587
588
589    assertNotNull("Dataset should be present", dataset);
590    assertEquals(organizationKey, dataset.getPublishingOrganizationKey());
591    assertEquals(installationKey, dataset.getInstallationKey());
592    assertEquals(DatasetType.METADATA, dataset.getType());
593    assertEquals(Requests.DATASET_NAME, dataset.getTitle());
594    assertEquals(new DOI(Requests.DOI), dataset.getDoi()); // ensure that we handle the parsing
595    assertEquals(Requests.DATASET_NAME_LANGUAGE, dataset.getLanguage().getIso2LetterCode());
596    assertEquals(Requests.DATASET_DESCRIPTION, dataset.getDescription());
597    assertEquals(Requests.DATASET_HOMEPAGE_URL, dataset.getHomepage().toString());
598    assertEquals(Requests.DATASET_LOGO_URL, dataset.getLogoUrl().toString());
599    assertNotNull(dataset.getCreated());
600    assertNotNull(dataset.getModified());
601
602    // check dataset's primary contact was properly persisted
603    Contact contact = dataset.getContacts().get(0);
604    assertNotNull("Dataset primary contact should be present", contact);
605    assertNotNull(contact.getKey());
606    assertTrue(contact.isPrimary());
607    assertEquals(Requests.DATASET_PRIMARY_CONTACT_NAME, contact.getFirstName());
608    assertEquals(Requests.DATASET_PRIMARY_CONTACT_EMAIL, contact.getEmail());
609    assertEquals(Requests.DATASET_PRIMARY_CONTACT_PHONE, contact.getPhone());
610    assertEquals(Requests.DATASET_PRIMARY_CONTACT_ADDRESS, contact.getAddress());
611    assertEquals(Requests.DATASET_PRIMARY_CONTACT_DESCRIPTION, contact.getDescription());
612    assertEquals(ContactType.ADMINISTRATIVE_POINT_OF_CONTACT, contact.getType());
613    assertNotNull(contact.getCreated());
614    assertNotNull(contact.getCreatedBy());
615    assertNotNull(contact.getModified());
616    assertNotNull(contact.getModifiedBy());
617
618    return dataset;
619  }

621  /**
622   * Retrieve dataset presumed already to exist, and make a series of assertions to ensure it is valid.
623   *
624   * @param dataset dataset
625   * @param organizationKey publishing organization key
626   * @param installationKey installation key
627   */
628  private void validateExistingDataset(Dataset dataset, UUID organizationKey, UUID installationKey) {
629    assertNotNull("Dataset should be present", dataset);
630    assertEquals(organizationKey, dataset.getPublishingOrganizationKey());
631    assertEquals(installationKey, dataset.getInstallationKey());
632    assertEquals(DatasetType.OCCURRENCE, dataset.getType());
633    // expected to change on update
634    assertNotEquals(Requests.DATASET_NAME, dataset.getTitle());
635    assertNotEquals(Requests.DATASET_DESCRIPTION, dataset.getDescription());
636    assertNotEquals(Requests.DATASET_HOMEPAGE_URL, dataset.getHomepage());
637    assertNotEquals(Requests.DATASET_LOGO_URL, dataset.getLogoUrl());
638    Date modified = dataset.getModified();
639    assertNotNull(modified);
640    String modifiedBy = dataset.getModifiedBy();
641    assertNotNull(modifiedBy);
642    assertTrue(dataset.getContacts().isEmpty());
643    assertTrue(dataset.getEndpoints().isEmpty());
644    // not expected to change
645    assertEquals(Datasets.DATASET_LANGUAGE, dataset.getLanguage());
646    assertEquals(Datasets.DATASET_RIGHTS, dataset.getRights());
647    assertEquals(Datasets.DATASET_CITATION.getIdentifier(), dataset.getCitation().getIdentifier());
648    assertEquals(Datasets.DATASET_ABBREVIATION, dataset.getAbbreviation());
649    assertEquals(Datasets.DATASET_ALIAS, dataset.getAlias());
650  }
