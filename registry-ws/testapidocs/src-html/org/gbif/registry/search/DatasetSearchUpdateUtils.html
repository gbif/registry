001package org.gbif.registry.search;
002
003import java.util.concurrent.TimeUnit;
004
005import com.google.common.base.Preconditions;
006import com.google.common.base.Stopwatch;
007import com.google.common.base.Throwables;
008import org.slf4j.Logger;
009import org.slf4j.LoggerFactory;
010
011
012public class DatasetSearchUpdateUtils {
013
014  // how often to poll and wait for SOLR to update
015  private static final int SOLR_UPDATE_TIMEOUT_SECS = 10;
016  private static final int SOLR_UPDATE_POLL_MSECS = 10;
017  private static final Logger LOG = LoggerFactory.getLogger(DatasetSearchUpdateUtils.class);
018
019  /**
020   * Waits for SOLR update threads to finish.
021   */
022  public static void awaitUpdates(DatasetIndexUpdateListener datasetIndexUpdater) {
023    Preconditions.checkNotNull(datasetIndexUpdater, "Index updater is required");
024    try {
025      Stopwatch stopWatch = Stopwatch.createStarted();
026      while (datasetIndexUpdater.queuedUpdates() > 0) {
027        Thread.sleep(SOLR_UPDATE_POLL_MSECS);
028        if (stopWatch.elapsed(TimeUnit.SECONDS) > SOLR_UPDATE_TIMEOUT_SECS) {
029          throw new IllegalStateException("Failing test due to unreasonable timeout on SOLR update");
030        }
031      }
032      LOG.info("Waited {} msecs for SOLR update backlog to clear successfully",
033        stopWatch.elapsed(TimeUnit.MILLISECONDS));
034
035    } catch (InterruptedException e) {
036      throw Throwables.propagate(e);
037    }
038  }
039}
