013package org.gbif.registry.utils;
014
015import org.gbif.api.model.common.DOI;
016import org.gbif.api.model.registry.Citation;
017import org.gbif.api.model.registry.Dataset;
018import org.gbif.api.service.registry.DatasetService;
019import org.gbif.api.vocabulary.Language;
020import org.gbif.registry.guice.RegistryTestModules;
021import org.gbif.registry.ws.resources.DatasetResource;
022
023import java.util.UUID;
024
025import com.google.inject.Injector;
026import org.codehaus.jackson.type.TypeReference;
027
028public class Datasets extends JsonBackedData<Dataset> {
029
030  private static final Datasets INSTANCE = new Datasets();
031  private static DatasetService datasetService;
032
033  public static final String DATASET_ALIAS = "BGBM";
034  public static final String DATASET_ABBREVIATION = "BGBM";
035  public static final Language DATASET_LANGUAGE = Language.DANISH;
036  public static final String DATASET_RIGHTS = "The rights";
037  public static final DOI DATASET_DOI = new DOI(DOI.TEST_PREFIX, "gbif.2014.XSD123");
038  public static final Citation DATASET_CITATION = new Citation("This is a citation text", "ABC");
039
040  public Datasets() {
041    super("data/dataset.json", new TypeReference<Dataset>() {
042    });
043    Injector i = RegistryTestModules.webservice();
044    datasetService = i.getInstance(DatasetResource.class);
045  }
046
047  public static Dataset newInstance(UUID publishingOrganizationKey, UUID installationKey) {
048    Dataset d = INSTANCE.newTypedInstance();
049    d.setPublishingOrganizationKey(publishingOrganizationKey);
050    d.setInstallationKey(installationKey);
051    d.setDoi(DATASET_DOI);
052    return d;
053  }
054
055  /**
056   * Persist a new Dataset associated to an publishing organization and installation for use in Unit Tests.
057   *
058   * @param organizationKey publishing organization key
059   * @param installationKey installation key
060   * @return persisted Dataset
061   */
062  public static Dataset newPersistedInstance(UUID organizationKey, UUID installationKey) {
063    Dataset dataset = Datasets.newInstance(organizationKey, installationKey);
064    UUID key = datasetService.create(dataset);
065    // some properties like created, modified are only set when the dataset is retrieved anew
066    return datasetService.get(key);
067  }
068
069}
