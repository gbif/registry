016package org.gbif.registry.events;
017
018import org.gbif.common.messaging.guice.PostalServiceModule;
019
020import java.util.Properties;
021
022import com.google.common.eventbus.EventBus;
023import com.google.inject.AbstractModule;
024import org.slf4j.Logger;
025import org.slf4j.LoggerFactory;
026
027/**
028 * One can either extend and bind listeners in this class, or bind them in other modules.
029 * Note that you MUST start the listener. Often the easiest way of doing this is simply to bind it as an eager singleton
030 * in guice.
031 */
032public class EventModule extends AbstractModule {
033  private static final Logger LOG = LoggerFactory.getLogger(EventModule.class);
034  public static final String MESSAGING_ENABLED_PROPERTY = "registry.postalservice.enabled";
035  public static final String EMBARGO_SECONDS_PROPERTY = "registry.postalservice.embargoSeconds";
036  private final EventBus eventBus = new EventBus();
037  private final Properties properties;
038
039  public EventModule(Properties properties) {
040    this.properties = properties;
041  }
042
043  @Override
044  protected void configure() {
045    bind(EventBus.class).toInstance(eventBus);
046    bindEventListeners();
047  }
048
049  /**
050   * Bind Event listeners here as eager Singletons.
051   */
052  private void bindEventListeners() {
053    if (properties.getProperty(MESSAGING_ENABLED_PROPERTY, "false").equals("true")) {
054      install(new PostalServiceModule("registry", properties));
055
056      // bind the embargo, defaulting to 0 seconds if there is none specified
057      try {
058        bind(Integer.class)
059          .annotatedWith(EmbargoDuration.class)
060          .toInstance(Integer.valueOf(properties.getProperty(EMBARGO_SECONDS_PROPERTY, "0")));
061      } catch (NumberFormatException e) {
062        LOG.error("{} is not of expected format. Expected single integer (e.g. 60), found {}",
063                  EMBARGO_SECONDS_PROPERTY, properties.getProperty(EMBARGO_SECONDS_PROPERTY));
064        throw e;
065      }
066
067      bind(MessageSendingEventListener.class).asEagerSingleton();
068    }
069  }
070}
