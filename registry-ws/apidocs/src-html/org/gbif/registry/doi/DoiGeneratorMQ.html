001package org.gbif.registry.doi;
002
003import org.gbif.api.model.common.DOI;
004import org.gbif.api.model.common.DoiData;
005import org.gbif.api.model.common.DoiStatus;
006import org.gbif.common.messaging.api.Message;
007import org.gbif.common.messaging.api.MessagePublisher;
008import org.gbif.common.messaging.api.messages.ChangeDoiMessage;
009import org.gbif.doi.metadata.datacite.DataCiteMetadata;
010import org.gbif.doi.service.InvalidMetadataException;
011import org.gbif.doi.service.datacite.DataCiteValidator;
012import org.gbif.registry.events.MessageSendingEventListener;
013import org.gbif.registry.persistence.mapper.DoiMapper;
014
015import java.io.IOException;
016import java.net.URI;
017import java.util.UUID;
018import javax.annotation.Nullable;
019
020import com.google.common.base.Preconditions;
021import com.google.common.base.Strings;
022import com.google.inject.Inject;
023import com.google.inject.name.Named;
024import org.apache.commons.lang3.RandomStringUtils;
025import org.apache.commons.lang3.exception.ExceptionUtils;
026import org.apache.ibatis.exceptions.PersistenceException;
027import org.slf4j.Logger;
028import org.slf4j.LoggerFactory;
029
030import static com.google.common.base.Preconditions.checkArgument;
031import static com.google.common.base.Preconditions.checkNotNull;
032
033public class DoiGeneratorMQ implements DoiGenerator {
034  private static final Logger LOG = LoggerFactory.getLogger(MessageSendingEventListener.class);
035
036  private final DoiMapper mapper;
037  private final URI datasetTarget;
038  private final URI downloadTarget;
039  private final String prefix;
040  private final static int RANDOM_LENGTH = 6;
041  /**
042   * The messagePublisher can be optional, and optional is not supported in constructor injection.
043   */
044  @Inject(optional = true)
045  private final MessagePublisher messagePublisher = null;
046
047  @Inject
048  public DoiGeneratorMQ(DoiMapper mapper, @Named("portal.url") URI portal, @Named("doi.prefix") String prefix) {
049    checkArgument(prefix.startsWith("10."), "DOI prefix must begin with 10.");
050    this.prefix = prefix;
051    this.mapper = mapper;
052    checkNotNull(portal, "portal base URL can't be null");
053    checkArgument(portal.isAbsolute(), "portal base URL must be absolute");
054    datasetTarget = portal.resolve("dataset/");
055    downloadTarget = portal.resolve("occurrence/download/");
056  }
057
058  private DOI newDOI(final String shoulder, DoiType type) {
059    // only try for hundred times then fail
060    for (int x=0; x<100; x++) {
061      DOI doi = random(shoulder);
062      try {
063        mapper.create(doi, type);
064        return doi;
065      } catch (PersistenceException e) {
066        // might have hit a unique constraint, try another doi
067        LOG.debug("Random {} DOI {} existed. Try another one", type, doi, e.getMessage());
068      }
069    }
070    throw new IllegalStateException("Tried 100 random DOIs and none worked, Giving up");
071  }
072
073  @Override
074  public DOI newDatasetDOI() {
075    return newDOI("", DoiType.DATASET);
076  }
077
078  @Override
079  public DOI newDownloadDOI() {
080    return newDOI("dl.", DoiType.DOWNLOAD);
081  }
082
083  @Override
084  public boolean isGbif(DOI doi) {
085    return doi != null && doi.getPrefix().equalsIgnoreCase(prefix);
086  }
087
088  @Override
089  public void failed(DOI doi, InvalidMetadataException e) {
090    // Updates the doi table to FAILED status and uses the error stacktrace as the xml for debugging.
091    mapper.update(doi, new DoiData(DoiStatus.FAILED, null), ExceptionUtils.getStackTrace(e));
092  }
093
094  /**
095   * @return a random DOI with the given prefix. It is not guaranteed to be unique and might exist already
096   */
097  private DOI random(@Nullable String shoulder) {
098    String suffix = Strings.nullToEmpty(shoulder) + RandomStringUtils.randomAlphanumeric(RANDOM_LENGTH);
099    return new DOI(prefix, suffix);
100  }
101
102  @Override
103  public void registerDataset(DOI doi, DataCiteMetadata metadata, UUID datasetKey) throws InvalidMetadataException {
104    Preconditions.checkNotNull(doi, "DOI required");
105    Preconditions.checkNotNull(datasetKey, "Dataset key required");
106    Preconditions.checkNotNull(messagePublisher,"No message publisher configured to send DoiChangeMessage");
107
108    String xml = DataCiteValidator.toXml(doi, metadata);
109    Message message = new ChangeDoiMessage(DoiStatus.REGISTERED, doi, xml, datasetTarget.resolve(datasetKey.toString()));
110
111    try {
112      messagePublisher.send(message, true);
113    } catch (IOException e) {
114      LOG.error("Failed sending DoiChangeMessage for {} and dataset {}", doi, datasetKey, e);
115    }
116
117  }
118
119  @Override
120  public void registerDownload(DOI doi, DataCiteMetadata metadata, String downloadKey) throws InvalidMetadataException {
121    Preconditions.checkNotNull(doi, "DOI required");
122    Preconditions.checkNotNull(downloadKey, "Download key required");
123    Preconditions.checkNotNull(messagePublisher,"No message publisher configured to send DoiChangeMessage");
124
125    String xml = DataCiteValidator.toXml(doi, metadata);
126    Message message = new ChangeDoiMessage(DoiStatus.REGISTERED, doi, xml, downloadTarget.resolve(downloadKey));
127
128    try {
129      messagePublisher.send(message, true);
130    } catch (IOException e) {
131      LOG.error("Failed sending DoiChangeMessage for {} and download {}", doi, downloadKey, e);
132    }
133
134  }
135
136  @Override
137  public void delete(DOI doi) {
138    Preconditions.checkNotNull(doi, "DOI required");
139    Preconditions.checkNotNull(messagePublisher,"No message publisher configured to send DoiChangeMessage");
140
141    Message message = new ChangeDoiMessage(DoiStatus.DELETED, doi, null, null);
142
143    try {
144      messagePublisher.send(message, true);
145    } catch (IOException e) {
146      LOG.error("Failed sending DoiChangeMessage for {}", doi, e);
147    }
148
149  }
150}
