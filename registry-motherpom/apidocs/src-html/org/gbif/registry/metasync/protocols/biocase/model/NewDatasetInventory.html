016package org.gbif.registry.metasync.protocols.biocase.model;
017
018import java.net.URI;
019import java.util.List;
020
021import com.google.common.base.Objects;
022import com.google.common.collect.Lists;
023import org.apache.commons.digester3.annotations.rules.BeanPropertySetter;
024import org.apache.commons.digester3.annotations.rules.ObjectCreate;
025import org.apache.commons.digester3.annotations.rules.SetNext;
026
027/**
028 * This object represents the data received when doing an inventory request using the new functionality as of BioCASe
029 * 3.4.
030 *
031 * @see <a href="http://wiki.bgbm.org/bps/index.php/VersionHistory#Version_3.4_.5B2013-02-20.5D">BioCASe Version
032 *      History</a>
033 */
034@ObjectCreate(pattern = "inventory")
035public class NewDatasetInventory {
036
037  @BeanPropertySetter(pattern = "inventory/status")
038  private String status;
039  @BeanPropertySetter(pattern = "inventory/service_url")
040  private URI serviceUrl;
041  private List<InventoryDataset> datasets = Lists.newArrayList();
042
043  public String getStatus() {
044    return status;
045  }
046
047  public void setStatus(String status) {
048    this.status = status;
049  }
050
051  public URI getServiceUrl() {
052    return serviceUrl;
053  }
054
055  public void setServiceUrl(URI serviceUrl) {
056    this.serviceUrl = serviceUrl;
057  }
058
059  public List<InventoryDataset> getDatasets() {
060    return datasets;
061  }
062
063  public void setDatasets(List<InventoryDataset> datasets) {
064    this.datasets = datasets;
065  }
066
067  @SetNext
068  public void addDataset(InventoryDataset dataset) {
069    datasets.add(dataset);
070  }
071
072  @Override
073  public String toString() {
074    return Objects.toStringHelper(this)
075      .add("status", status)
076      .add("serviceUrl", serviceUrl)
077      .add("datasets", datasets)
078      .toString();
079  }
080
081}
