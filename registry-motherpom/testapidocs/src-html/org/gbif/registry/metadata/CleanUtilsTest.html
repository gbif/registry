001package org.gbif.registry.metadata;
002
003import org.gbif.api.model.registry.Citation;
004import org.gbif.api.model.registry.Contact;
005
006import com.google.common.collect.Lists;
007import org.junit.Test;
008
009import static org.junit.Assert.assertEquals;
010import static org.junit.Assert.assertNull;
011
012public class CleanUtilsTest {
013
014  @Test
015  public void testRemoveEmptyStrings() throws Exception {
016    Contact c = new Contact();
017    c.setFirstName("");
018    c.setLastName("  ");
019    c.setCity("Berlin");
020    c.setPhone(Lists.newArrayList("12345"));
021    c.addEmail("   ");
022    c.addEmail("   my email");
023    c.addEmail(null);
024
025    CleanUtils.removeEmptyStrings(c);
026
027    assertEquals("Berlin", c.getCity());
028    assertEquals(1, c.getEmail().size());
029    assertEquals("   my email", c.getEmail().get(0));
030    assertNull(c.getFirstName());
031    assertNull(c.getLastName());
032    assertNull(c.getCountry());
033  }
034
035  @Test
036  public void testRemoveEmptyStringsCitation() throws Exception {
037    Citation c = new Citation();
038    c.setText("");
039    CleanUtils.removeEmptyStrings(c);
040
041    assertNull(c.getText());
042    assertNull(c.getIdentifier());
043  }
044}
