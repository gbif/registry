025package org.gbif.registry;
026
027import org.gbif.api.model.registry.Comment;
028import org.gbif.api.model.registry.NetworkEntity;
029import org.gbif.api.service.registry.CommentService;
030import org.gbif.registry.utils.Comments;
031
032import java.util.List;
033
034import static org.gbif.registry.LenientAssert.assertLenientEquals;
035
036import static org.junit.Assert.assertEquals;
037import static org.junit.Assert.assertNotNull;
038import static org.junit.Assert.assertTrue;
039
040public class CommentTests {
041
042  public static <T extends NetworkEntity> void testAddDelete(CommentService service, T entity) {
043
044    // check there are none on a newly created entity
045    List<Comment> comments = service.listComments(entity.getKey());
046    assertNotNull("Comment list should be empty, not null when no comments exist", comments);
047    assertTrue("Comment should be empty when none added", comments.isEmpty());
048
049    // test additions
050    service.addComment(entity.getKey(), Comments.newInstance());
051    service.addComment(entity.getKey(), Comments.newInstance());
052    comments = service.listComments(entity.getKey());
053    assertNotNull(comments);
054    assertEquals("2 commentss have been added", 2, comments.size());
055
056    // test deletion, ensuring correct one is deleted
057    service.deleteComment(entity.getKey(), comments.get(0).getKey());
058    comments = service.listComments(entity.getKey());
059    assertNotNull(comments);
060    assertEquals("1 comment should remain after the deletion", 1, comments.size());
061    Comment expected = Comments.newInstance();
062    Comment created = comments.get(0);
063    assertLenientEquals("Created entity does not read as expected", expected, created);
064  }
065
066}
