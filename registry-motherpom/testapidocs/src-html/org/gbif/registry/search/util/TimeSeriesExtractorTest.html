001package org.gbif.registry.search.util;
002
003import org.gbif.api.model.registry.eml.temporal.DateRange;
004import org.gbif.api.model.registry.eml.temporal.TemporalCoverage;
005
006import java.util.Calendar;
007import java.util.List;
008
009import com.google.common.collect.Lists;
010import org.junit.Test;
011
012import static org.junit.Assert.assertEquals;
013
014public class TimeSeriesExtractorTest {
015
016  @Test
017  public void testExtractDecades() throws Exception {
018
019    TimeSeriesExtractor ex = new TimeSeriesExtractor(1000, 2500, 1800, 2050);
020
021    List<TemporalCoverage> coverages = Lists.newArrayList();
022    coverages.add(range(1977, 1995));
023    coverages.add(range(1914, 1917));
024    coverages.add(range(987, 999));
025    coverages.add(range(2222, 3333));
026
027    List<Integer> periods = ex.extractDecades(coverages);
028    List<Integer> expected = Lists.newArrayList(1000, 1910, 1970, 1980, 1990, 2200, 2300, 2400, 2500);
029    assertEquals(expected, periods);
030  }
031
032  @Test
033  public void testSwappedDecades() throws Exception {
034
035    TimeSeriesExtractor ex = new TimeSeriesExtractor(1000, 2500, 1800, 2050);
036
037    List<TemporalCoverage> coverages = Lists.newArrayList();
038    coverages.add(range(1995, 1977));
039    coverages.add(range(1914, 1917));
040    coverages.add(range(987, 999));
041    coverages.add(range(3333, 2222));
042
043    List<Integer> periods = ex.extractDecades(coverages);
044    List<Integer> expected = Lists.newArrayList(1000, 1910);
045    assertEquals(expected, periods);
046  }
047
048  private DateRange range(int year1, int year2) {
049    DateRange dr = new DateRange();
050
051    Calendar cal = Calendar.getInstance();
052    cal.set(Calendar.YEAR, year1);
053    dr.setStart(cal.getTime());
054
055    cal.set(Calendar.YEAR, year2);
056    dr.setEnd(cal.getTime());
057
058    return dr;
059  }
060}
