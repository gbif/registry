052  @Mock
053  private HttpClient client;
054  private BiocaseMetadataSynchroniser synchroniser;
055  private Installation installation;
056
057  @Before
058  public void setup() {
059    synchroniser = new BiocaseMetadataSynchroniser(client);
060    installation = new Installation();
061    installation.setType(InstallationType.BIOCASE_INSTALLATION);
062    Endpoint endpoint = new Endpoint();
063    endpoint.setUrl(URI.create("http://localhost"));
064    installation.addEndpoint(endpoint);
065  }
066
067  @Test
068  public void testCanHandle() {
069    installation.setType(InstallationType.DIGIR_INSTALLATION);
070    assertThat(synchroniser.canHandle(installation)).isFalse();
071
072    installation.setType(InstallationType.BIOCASE_INSTALLATION);
073    assertThat(synchroniser.canHandle(installation)).isTrue();
074  }
075
076  /**
077   * This tests a BioCASe endpoint that supports the old style inventory and ABCD 2.06
078   */
079  @Test
080  public void testAddedDataset1() throws Exception {
081    when(client.execute(any(HttpGet.class))).thenReturn(prepareResponse(200, "biocase/capabilities1.xml"))
082      .thenReturn(prepareResponse(200, "biocase/inventory1.xml"))
083      .thenReturn(prepareResponse(200, "biocase/dataset1.xml"));
084    SyncResult syncResult = synchroniser.syncInstallation(installation, new ArrayList<Dataset>());
085    assertThat(syncResult.exception).isNull();
086    assertThat(syncResult.deletedDatasets).isEmpty();
087    assertThat(syncResult.existingDatasets).isEmpty();
088    assertThat(syncResult.addedDatasets).hasSize(1);
089
090    Dataset dataset = syncResult.addedDatasets.get(0);
091    assertThat(dataset.getTitle()).isEqualTo("Pontaurus");
092    assertThat(dataset.getCitation().getText()).isEqualTo("All credit to Markus Doring");
093    assertThat(dataset.getDoi()).isNull();
094    // endpoints
095    assertThat(dataset.getEndpoints().size()).isEqualTo(1);
096    assertThat(dataset.getEndpoints().get(0).getType()).isEqualTo(EndpointType.BIOCASE);
097  }

165    Dataset dataset = new Dataset();
166    dataset.setTitle("Pontaurus");

168    when(client.execute(any(HttpGet.class))).thenReturn(prepareResponse(200, "biocase/capabilities1.xml"))
169      .thenReturn(prepareResponse(200, "biocase/inventory1.xml"))

172    assertThat(syncResult.deletedDatasets).describedAs("Deleted datasets").isEmpty();
173    assertThat(syncResult.existingDatasets).hasSize(1);
174    assertThat(syncResult.addedDatasets).isEmpty();
175
176    assertThat(syncResult.existingDatasets.get(dataset).getTitle()).isEqualTo("Pontaurus");

179  /**
180   * Prepares a {@link HttpResponse} with the given response status and the content of the file.
181   */
182  private HttpResponse prepareResponse(int responseStatus, String fileName) throws IOException {
183    HttpResponse response =
184      new BasicHttpResponse(new BasicStatusLine(new ProtocolVersion("HTTP", 1, 1), responseStatus, ""));
185    response.setStatusCode(responseStatus);
186    byte[] bytes = Resources.toByteArray(Resources.getResource(fileName));
187    response.setEntity(new ByteArrayEntity(bytes));
188    return response;
189  }
