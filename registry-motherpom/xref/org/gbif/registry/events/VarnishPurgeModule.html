5   import java.net.URI;
6   import java.util.Properties;
7   
8   import com.google.inject.PrivateModule;
9   import org.apache.commons.lang3.StringUtils;
10  import org.apache.http.impl.client.CloseableHttpClient;
11  import org.slf4j.Logger;
12  import org.slf4j.LoggerFactory;
13  
14  /**
15   * A Guice module that sets up implementation of the varnish purging via the event bus.
16   * If no or a bad api.cache.purge.url property is configured the listener will not be wired up.
17   */
18  public class VarnishPurgeModule extends PrivateModule {
19  
20    public static final String API_CACHE_PURGE_URL_PROPERTY = "api.cache.purge.url";
21  
22    private static final Logger LOG = LoggerFactory.getLogger(VarnishPurgeModule.class);
23    private static final int DEFAULT_HTTP_TIMEOUT_MSECS = 2000;
24    private final int httpThreads;
25    private final URI apiRoot;
26  
27    public VarnishPurgeModule(Properties properties) {
28      String apiCachePurgeUrl = properties.getProperty(API_CACHE_PURGE_URL_PROPERTY);
29  
30      try {
31        if(StringUtils.isNotBlank(apiCachePurgeUrl)) {
32          apiRoot = URI.create(apiCachePurgeUrl);
33        }
34        else{
35          apiRoot = null;
36        }
37        httpThreads = Integer.valueOf(properties.getProperty("purging.threads", "100").trim());
38      } catch (RuntimeException e) {
39        LOG.error("Failed to initialize varnish purger because of invalid properties", e.getMessage());
40        throw e;
41      }
42    }
43  
44    @Override
45    protected void configure() {
46      if(apiRoot == null){
47        LOG.warn("No varnish purging configured: {} not found or empty", API_CACHE_PURGE_URL_PROPERTY);
48        return;
49      }
50      if (httpThreads > 0) {
51        bind(URI.class).toInstance(apiRoot);
52        bind(CloseableHttpClient.class).toInstance(HttpUtil.newMultithreadedClient(
53          DEFAULT_HTTP_TIMEOUT_MSECS, httpThreads, httpThreads));
54        bind(VarnishPurgeListener.class).asEagerSingleton();
55        LOG.info("Varnish purging enabled with {} threads and API {}", httpThreads, apiRoot);
56      } else {
57        LOG.warn("No varnish purging configured. Please set purging.threads greater than zero if you want it");
58      }
59    }
60  
61  }
