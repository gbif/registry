49  /**
50   * A MyBATIS implementation of the service.
51   */
52  @Path("network")
53  @Singleton
54  public class NetworkResource extends BaseNetworkEntityResource<Network> implements NetworkService {
55  
56    private final DatasetMapper datasetMapper;
57    private final NetworkMapper networkMapper;
58  
59    @Inject
60    public NetworkResource(
61      NetworkMapper networkMapper,
62      ContactMapper contactMapper,
63      EndpointMapper endpointMapper,
64      IdentifierMapper identifierMapper,
65      MachineTagMapper machineTagMapper,
66      TagMapper tagMapper,
67      CommentMapper commentMapper,
68      DatasetMapper datasetMapper,
69      EventBus eventBus,
70      EditorAuthorizationService userAuthService) {
71      super(networkMapper,
72        commentMapper,
73        contactMapper,
74        endpointMapper,
75        identifierMapper,
76        machineTagMapper,
77        tagMapper,
78        Network.class,
79        eventBus,
80        userAuthService);
81      this.datasetMapper = datasetMapper;
82      this.networkMapper = networkMapper;
83    }
84  
85  
86    /**
87     * All network entities support simple (!) search with "&q=".
88     * This is to support the console user interface, and is in addition to any complex, faceted search that might
89     * additionally be supported, such as dataset search.
90     */
91    @GET
92    public PagingResponse<Network> list(@Nullable @QueryParam("q") String query,
93      @Nullable @QueryParam("identifierType") IdentifierType identifierType,
94      @Nullable @QueryParam("identifier") String identifier,
95      @Nullable @Context Pageable page) {
96      // This is getting messy: http://dev.gbif.org/issues/browse/REG-426
97      if (identifierType != null && identifier != null) {
98        return listByIdentifier(identifierType, identifier, page);
99      } else if (identifier != null) {
100       return listByIdentifier(identifier, page);
101     } else if (Strings.isNullOrEmpty(query)) {
102       return list(page);
103     } else {
104       return search(query, page);
105     }
106   }
107 
108   @Path("{key}/constituents")
109   @GET
110   @Override
111   public PagingResponse<Dataset> listConstituents(@PathParam("key") UUID networkKey, @Context Pageable page) {
112     return pagingResponse(page, (long) networkMapper.countDatasetsInNetwork(networkKey),
113       datasetMapper.listDatasetsInNetwork(networkKey, page));
114   }
115 
116   @Path("{key}/constituents/{datasetKey}")
117   @POST
118   @RolesAllowed(ADMIN_ROLE)
119   @Override
120   public void addConstituent(@PathParam("key") UUID networkKey, @PathParam("datasetKey") UUID datasetKey) {
121     networkMapper.addDatasetConstituent(networkKey, datasetKey);
122   }
123 
124 
125   @Path("{key}/constituents/{datasetKey}")
126   @DELETE
127   @RolesAllowed(ADMIN_ROLE)
128   @Override
129   public void removeConstituent(@PathParam("key") UUID networkKey, @PathParam("datasetKey") UUID datasetKey) {
130     networkMapper.deleteDatasetConstituent(networkKey, datasetKey);
131   }
132 
133 }
