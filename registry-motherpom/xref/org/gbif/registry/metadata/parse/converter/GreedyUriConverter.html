1   package org.gbif.registry.metadata.parse.converter;
2   
3   import java.net.URI;
4   import java.net.URL;
5   
6   import com.google.common.base.CharMatcher;
7   import com.google.common.base.Preconditions;
8   import com.google.common.base.Strings;
9   import org.apache.commons.beanutils.converters.AbstractConverter;
10  
11  /**
12   * Greedy String to URI converter. Greedy in the sense that it will try to add a default protocol (http://) in
13   * cases where none exists.
14   * For example, this captures URLs such as www.gbif.org that otherwise would get converted into a wrong URI.
15   */
16  public final class GreedyUriConverter extends AbstractConverter {
17  
18    public static final String DEFAULT_PROTOCOL = "http://";
19  
20    /**
21     * Construct a <b>URI</b> <i>Converter</i> that throws
22     * a {@code ConversionException} if an error occurs.
23     */
24    public GreedyUriConverter() {
25    }
26  
27    /**
28     * Construct a <b>URI</b> <i>Converter</i> that returns
29     * a default value if an error occurs.
30     *
31     * @param defaultValue The default value to be returned
32     *                     if the value to be converted is missing or an error
33     *                     occurs converting the value.
34     */
35    public GreedyUriConverter(Object defaultValue) {
36      super(defaultValue);
37    }
38  
39    /**
40     * Convert a String into a java.net.URI. In case its missing the protocol prefix, it is prefixed with the default
41     * protocol.
42     *
43     * @param type  Data type to which this value should be converted.
44     * @param value The input value to be converted.
45     *
46     * @return The converted value, or null if an exception occurred
47     */
48    @Override
49    protected Object convertToType(Class type, Object value) {
50      Preconditions.checkNotNull(value, "Must provide a type to convert to null");
51      String valueAsString = Strings.emptyToNull(CharMatcher.WHITESPACE.trimFrom(String.valueOf(value)));
52  
53      URI uri = null;
54      if (!Strings.isNullOrEmpty(valueAsString)) {
55        try {
56          uri = URI.create(valueAsString);
57          // try adding the default scheme if the URI is opaque or missing
58          if (uri.getScheme() == null || uri.isOpaque()) {
59            try {
60              uri = URI.create(DEFAULT_PROTOCOL + valueAsString);
61            } catch (IllegalArgumentException e) {
62              // keep the previous scheme-less result
63            }
64          }
65        } catch (IllegalArgumentException e) {
66        }
67      }
68      return uri;
69    }
70  
71    /**
72     * Return the default type this {@code Converter} handles.
73     *
74     * @return The default type this {@code Converter} handles.
75     */
76    @Override
77    protected Class getDefaultType() {
78      return URI.class;
79    }
80  
81    /**
82     * Convert a String into a java.net.URI. In case its missing the protocol prefix, it is prefixed with the default
83     * protocol.
84     *
85     * @param value The input value to be converted.
86     *
87     * @return The converted value, or null if an exception occurred
88     */
89    public URI convert(String value) {
90      return (URI) convertToType(URL.class, value);
91    }
92  
93  }
