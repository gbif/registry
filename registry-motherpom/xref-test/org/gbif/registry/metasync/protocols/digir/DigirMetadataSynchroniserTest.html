53  @RunWith(MockitoJUnitRunner.class)
54  public class DigirMetadataSynchroniserTest {
55  
56    @Mock
57    private HttpClient client;
58    private DigirMetadataSynchroniser synchroniser;
59    private Installation installation;
60  
61    @Before
62    public void setup() {
63      synchroniser = new DigirMetadataSynchroniser(client);
64  
65      installation = new Installation();
66      installation.setType(InstallationType.DIGIR_INSTALLATION);
67      Endpoint endpoint = new Endpoint();
68      endpoint.setUrl(URI.create("http://localhost"));
69      installation.addEndpoint(endpoint);
70    }
71  
72    @Test
73    public void testCanHandle() {
74      installation.setType(InstallationType.BIOCASE_INSTALLATION);
75      assertThat(synchroniser.canHandle(installation)).isFalse();
76  
77      installation.setType(InstallationType.DIGIR_INSTALLATION);
78      assertThat(synchroniser.canHandle(installation)).isTrue();
79    }
80  
81    /**
82     * A simple test to see if multiple datasets are parsed successfully.
83     */
84    @Test
85    public void testAddedDatasets() throws Exception {
86      when(client.execute(any(HttpGet.class))).thenReturn(prepareResponse(200, "digir/test1.xml"));
87      SyncResult syncResult = synchroniser.syncInstallation(installation, new ArrayList<Dataset>());
88      assertThat(syncResult.deletedDatasets).isEmpty();
89      assertThat(syncResult.existingDatasets).isEmpty();
90      assertThat(syncResult.addedDatasets).hasSize(8);
91    }
92  
93    /**
94     * This tests a Metadata response with just one resources but validates more things from this parsed Dataset.
95     */
96    @Test
97    public void testAddedDataset() throws Exception {
98      when(client.execute(any(HttpGet.class))).thenReturn(prepareResponse(200, "digir/test2.xml"));
99      SyncResult syncResult = synchroniser.syncInstallation(installation, new ArrayList<Dataset>());
100     assertThat(syncResult.deletedDatasets).isEmpty();
101     assertThat(syncResult.existingDatasets).isEmpty();
102     assertThat(syncResult.addedDatasets).hasSize(1);
103 
104     Dataset dataset = syncResult.addedDatasets.get(0);
105     assertThat(dataset.getTitle()).isEqualTo("Distribution of benthic foraminifera of sediment core PS1388-3");
106     assertThat(dataset.getHomepage()).isEqualTo(URI.create("http://doi.pangaea.de/doi:10.1594/PANGAEA.51131"));
107     assertThat(dataset.getCitation().getText()).isEqualTo(
108       "Mackensen, Andreas; Grobe, Hannes; Hubberten, Hans-Wolfgang; Spieß, Volkhard; Fütterer, Dieter K (1989): Distribution of benthic foraminifera of sediment core PS1388-3, doi:10.1594/PANGAEA.51131");
109     assertThat(dataset.getIdentifiers().size()).isEqualTo(0); // verify that we don't create an identifier for the DOI
110     assertThat(dataset.getDoi().getDoiName()).isEqualTo("10.1594/pangaea.51131");
111     // endpoints
112     assertThat(dataset.getEndpoints().size()).isEqualTo(1);
113     assertThat(dataset.getEndpoints().get(0).getType()).isEqualTo(EndpointType.DIGIR);
114   }

116   @Test
117   public void testDeletedDataset() throws Exception {
118     Dataset dataset = new Dataset();
119     dataset.addMachineTag(MachineTag.newInstance(Constants.METADATA_NAMESPACE, Constants.DIGIR_CODE, "foobar"));
120     dataset.setTitle("Foobar");
121 
122     when(client.execute(any(HttpGet.class))).thenReturn(prepareResponse(200, "digir/test2.xml"));
123     SyncResult syncResult = synchroniser.syncInstallation(installation, Lists.newArrayList(dataset));
124     assertThat(syncResult.deletedDatasets).hasSize(1);
125     assertThat(syncResult.existingDatasets).isEmpty();
126     assertThat(syncResult.addedDatasets).hasSize(1);
127 
128     assertThat(syncResult.deletedDatasets.get(0).getTitle()).isEqualTo("Foobar");
129   }

131   @Test
132   public void testUpdatedDataset() throws Exception {
133     Dataset dataset = new Dataset();
134     dataset.addMachineTag(MachineTag.newInstance(Constants.METADATA_NAMESPACE,
135                                                  Constants.DIGIR_CODE,
136                                                  "doi:10.1594/PANGAEA.51131"));
137     dataset.setTitle("Foobar");
138 
139     when(client.execute(any(HttpGet.class))).thenReturn(prepareResponse(200, "digir/test2.xml"));
140     SyncResult syncResult = synchroniser.syncInstallation(installation, Lists.newArrayList(dataset));
141     assertThat(syncResult.deletedDatasets).describedAs("Deleted datasets").isEmpty();
142     assertThat(syncResult.existingDatasets).hasSize(1);
143     assertThat(syncResult.addedDatasets).isEmpty();
144 
145     assertThat(syncResult.existingDatasets.get(dataset).getTitle()).isEqualTo(
146       "Distribution of benthic foraminifera of sediment core PS1388-3");
147   }

149   /**
150    * Make sure the determination of DiGIR endpoint type is catching DIGIR_MANIS.
151    */
152   @Test
153   public void testDetermineEndpointType() {
154     // populate map with namespace (conceptualSchema) / schemaLocation key value pair
155     Map<String, URI> schemas = Maps.newHashMap();
156     schemas.put("http://digir.net/schema/conceptual/darwin/2003/1.0",
157       URI.create("http://bnhm.berkeley.museum/manis/DwC/darwin2jrw030315.xsd"));
158     assertThat(synchroniser.determineEndpointType(schemas)).isEqualTo(EndpointType.DIGIR_MANIS);
159     // reset, try another
160     schemas.clear();
161     schemas.put("http://digir.net/schema/conceptual/darwin/2003/1.0",
162       URI.create("http://bnhm.berkeley.edu/DwC/bnhm_dc2_schema.xsd"));
163     assertThat(synchroniser.determineEndpointType(schemas)).isEqualTo(EndpointType.DIGIR_MANIS);
164   }

166   public HttpResponse prepareResponse(int responseStatus, String fileName) throws IOException {
167     HttpResponse response =
168       new BasicHttpResponse(new BasicStatusLine(new ProtocolVersion("HTTP", 1, 1), responseStatus, ""));
169     response.setStatusCode(responseStatus);
170     byte[] bytes = Resources.toByteArray(Resources.getResource(fileName));
171     response.setEntity(new ByteArrayEntity(bytes));
172     return response;
173   }
