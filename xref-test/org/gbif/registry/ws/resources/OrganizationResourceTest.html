8   public class OrganizationResourceTest {
9   
10    /**
11     * Ensure randomly generated password has size greater than or equal to MINIMUM_PASSWORD_SIZE and less than
12     * MAXIMUM_PASSWORD_SIZE, and that subsequent passwords being generated are in fact unique.
13     */
14    @Test
15    public void testGeneratePassword() {
16      String password1 = OrganizationResource.generatePassword();
17      assertTrue(password1.length() >= OrganizationResource.MINIMUM_PASSWORD_SIZE);
18      assertTrue(password1.length() < OrganizationResource.MAXIMUM_PASSWORD_SIZE);
19  
20      String password2 = OrganizationResource.generatePassword();
21      assertTrue(password2.length() >= OrganizationResource.MINIMUM_PASSWORD_SIZE);
22      assertTrue(password2.length() < OrganizationResource.MAXIMUM_PASSWORD_SIZE);
23  
24      String password3 = OrganizationResource.generatePassword();
25      assertTrue(password3.length() >= OrganizationResource.MINIMUM_PASSWORD_SIZE);
26      assertTrue(password3.length() < OrganizationResource.MAXIMUM_PASSWORD_SIZE);
27  
28      String password4 = OrganizationResource.generatePassword();
29      assertTrue(password4.length() >= OrganizationResource.MINIMUM_PASSWORD_SIZE);
30      assertTrue(password4.length() < OrganizationResource.MAXIMUM_PASSWORD_SIZE);
31  
32      assertNotEquals(password1, password2);
33      assertNotEquals(password1, password3);
34      assertNotEquals(password1, password4);
35  
36      assertNotEquals(password2, password3);
37      assertNotEquals(password2, password4);
38  
39      assertNotEquals(password3, password4);
40    }
41  }
