3   import java.util.Calendar;
4   import java.util.Date;
5   
6   import org.junit.Test;
7   
8   import static org.junit.Assert.assertEquals;
9   import static org.junit.Assert.assertNull;
10  
11  public class DateConverterTest {
12    private DateConverter converter = new DateConverter();
13  
14    @Test
15    public void parseDates() throws Throwable {
16      assertLang(2000, 10, 20, "2000-10-20");
17      assertLang(2000, 10, 20, "20.10.2000");
18    }
19  
20    private void assertLang(Integer year, Integer month, Integer day, String value) throws Throwable {
21      Date d = (Date) converter.convertToType(Date.class, value);
22  
23      Calendar cal = Calendar.getInstance();
24      cal.setTime(d);
25  
26      if (year != null) {
27        assertEquals(year, (Integer) cal.get(Calendar.YEAR));
28      }
29      if (month != null) {
30        assertEquals(month, (Integer) (1 + cal.get(Calendar.MONTH)));
31      }
32      if (day != null) {
33        assertEquals(day, (Integer) cal.get(Calendar.DAY_OF_MONTH));
34      }
35      if (year == null && month == null && day==null) {
36        assertNull(d);
37      }
38    }
39  }
