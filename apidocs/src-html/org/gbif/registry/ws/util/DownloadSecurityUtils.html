001package org.gbif.registry.ws.util;
002
003import org.gbif.api.model.occurrence.Download;
004import org.gbif.api.model.registry.DatasetOccurrenceDownloadUsage;
005
006import java.util.Collection;
007
008import javax.ws.rs.WebApplicationException;
009import javax.ws.rs.core.Response;
010import javax.ws.rs.core.SecurityContext;
011
012import org.slf4j.Logger;
013import org.slf4j.LoggerFactory;
014
015import static org.gbif.registry.ws.security.UserRoles.ADMIN_ROLE;
016
017/**
018 * Class that exposes common functions used by services that expose data about occurrence downloads.
019 */
020public class DownloadSecurityUtils {
021
022  private static final Logger LOG = LoggerFactory.getLogger(DownloadSecurityUtils.class);
023
024  private DownloadSecurityUtils() {
025    // private constructor
026  }
027
028  /**
029   * Checks if the user has the ADMIN_ROLE or is the same user in the current context.
030   */
031  public static void checkUserIsInSecurityContext(String user, SecurityContext securityContext) {
032    // A null securityContext means that the class is executed locally
033    if (!isUserAuthorizedInContext(securityContext, user)) {
034      LOG.warn(String.format("Unauthorized access detected, authenticated use, requested user %s", user));
035      throw new WebApplicationException(Response.Status.UNAUTHORIZED);
036    }
037  }
038
039  /**
040   * Checks if the user has the ADMIN_ROLE or is the same user in the current context.
041   */
042  public static boolean isUserAuthorizedInContext(SecurityContext securityContext, String user) {
043    return (securityContext == null || securityContext.isUserInRole(ADMIN_ROLE) || (securityContext.getUserPrincipal() != null
044    && securityContext.getUserPrincipal().getName().equals(user)));
045  }
046
047  /**
048   * Remove data that shouldn't be publicly exposed.
049   */
050  public static void clearSensitiveData(SecurityContext securityContext, Download download) {
051    if (download.getRequest() != null
052      && !isUserAuthorizedInContext(securityContext, download.getRequest().getCreator())) {
053      download.getRequest().setNotificationAddresses(null);
054    }
055  }
056
057  /**
058   * Remove data that shouldn't be publicly exposed.
059   */
060  public static void clearSensitiveData(SecurityContext securityContext,
061    Collection<DatasetOccurrenceDownloadUsage> downloadUsages) {
062    for (DatasetOccurrenceDownloadUsage downloadUsage : downloadUsages) {
063      clearSensitiveData(securityContext, downloadUsage.getDownload());
064    }
065  }
066}
