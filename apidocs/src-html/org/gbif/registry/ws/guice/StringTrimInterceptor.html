013package org.gbif.registry.ws.guice;
014
015import org.gbif.api.model.registry.Dataset;
016
017import java.lang.annotation.Annotation;
018
019import com.google.common.annotations.VisibleForTesting;
020import com.google.common.base.Objects;
021import com.google.common.base.Strings;
022import com.google.inject.AbstractModule;
023import com.google.inject.Module;
024import com.google.inject.matcher.Matchers;
025import org.aopalliance.intercept.MethodInterceptor;
026import org.aopalliance.intercept.MethodInvocation;
027import org.apache.bval.guice.ValidationModule;
028import org.apache.commons.beanutils.DynaClass;
029import org.apache.commons.beanutils.DynaProperty;
030import org.apache.commons.beanutils.WrapDynaBean;
031import org.slf4j.Logger;
032import org.slf4j.LoggerFactory;
033
034/**
035 * An interceptor that will trim all possible strings of a bean.
036 * All top level string properties are handled, as are those of nested objects that are in the GBIF registry model
037 * package.
038 * This will recurse only 5 levels deep, to guard against potential circular looping.
039 */
040public class StringTrimInterceptor implements MethodInterceptor {
041
042  private static final Logger LOG = LoggerFactory.getLogger(StringTrimInterceptor.class);
043  private static final int MAX_RECURSION = 5; // only goes 5 levels deep to stop potential circular loops
044
045
046  /**
047   * Sets up method level interception for those methods annotated with {@link Trim}.
048   * Normally this would go before any {@link ValidationModule}, so Strings are trimmed before validated.
049   */
050  public static Module newMethodInterceptingModule() {
051    return new AbstractModule() {
052
053      @Override
054      protected void configure() {
055        MethodInterceptor trimMethodInterceptor = new StringTrimInterceptor();
056        this.binder().requestInjection(trimMethodInterceptor);
057        this.bindInterceptor(Matchers.any(), Matchers.annotatedWith(Trim.class), trimMethodInterceptor);
058      }
059    };
060  }
061
062  @Override
063  public Object invoke(MethodInvocation invocation) throws Throwable {
064    Trim validate = invocation.getMethod().getAnnotation(Trim.class); // ensure it is annotated
065    if (validate != null) {
066      Annotation[][] paramAnnotations = invocation.getMethod().getParameterAnnotations();
067      for (int i = 0; i < paramAnnotations.length; i++) {
068        for (Annotation a : paramAnnotations[i]) {
069          if (Trim.class.isAssignableFrom(a.annotationType())) {
070            trimStringsOf(invocation.getArguments()[i]);
071          }
072        }
073      }
074    }
075    return invocation.proceed();
076  }
077
078  @VisibleForTesting
079  void trimStringsOf(Object target) {
080    trimStringsOf(target, 0);
081  }
082
083  private void trimStringsOf(Object target, int level) {
084    if (target != null && level <= MAX_RECURSION) {
085      LOG.debug("Trimming class: {}", target.getClass());
086
087      WrapDynaBean wrapped = new WrapDynaBean(target);
088      DynaClass dynaClass = wrapped.getDynaClass();
089      for (DynaProperty dynaProp : dynaClass.getDynaProperties()) {
090        if (String.class.isAssignableFrom(dynaProp.getType())) {
091          String prop = dynaProp.getName();
092          String orig = (String) wrapped.get(prop);
093          if (orig != null) {
094            String trimmed = Strings.emptyToNull(orig.trim());
095            if (!Objects.equal(orig, trimmed)) {
096              LOG.debug("Overriding value of [{}] from [{}] to [{}]", prop, orig, trimmed);
097              wrapped.set(prop, trimmed);
098            }
099          }
100        } else {
101          try {
102            // trim everything in the registry model package (assume that Dataset resides in the correct package here)
103            Object property = wrapped.get(dynaProp.getName());
104            if (property != null && Dataset.class.getPackage() == property.getClass().getPackage()) {
105              trimStringsOf(property, level + 1);
106            }
107
108          } catch (IllegalArgumentException e) {
109            // expected for non accessible properties
110          }
111        }
112      }
113    }
114  }
115}
