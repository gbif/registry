044/**
045 * The Registry WS module.
046 */
047public class RegistryWsServletListener extends GbifServletListener {
048
049  public static final String APP_CONF_FILE = "registry.properties";
050
051  private static final String API_URL_PROPERTY = "api.url";
052  // fail fast by designed, see CubeService usage
053  private static final String METRICS_WS_HTTP_TIMEOUT = "100";
054
055  public static final List<Class<? extends ContainerRequestFilter>> requestFilters = Lists.newArrayList();
056  public static final List<Class<? extends ContainerResponseFilter>> responseFilters = Lists.newArrayList();
057
058  static {
059    requestFilters.add(LegacyAuthorizationFilter.class);
060    requestFilters.add(EditorAuthorizationFilter.class);
061    responseFilters.add(AuthResponseCodeOverwriteFilter.class);
062  }
063
064  private static final String PACKAGES = "org.gbif.registry.ws.resources, org.gbif.registry.ws.provider, org.gbif.registry.oaipmh";
065
066  /**
067   * Get a subset of properties related to metrics.
068   * Uses the api.url to fill metrics.ws.url and set a small timeout (100 ms) for http
069   * This methods only exists because {@link org.gbif.ws.client.guice.GbifWsClientModule} uses Names.bindProperties(binder(), properties);
070   * which would lead to multiple bindings for the same property.
071   *
072   * @param properties
073   * @return
074   */
075  private Properties getMetricsProperties(Properties properties){
076    Properties metricsProperties = new Properties();
077    metricsProperties.setProperty("metrics.ws.url", properties.getProperty(API_URL_PROPERTY));
078    metricsProperties.setProperty(MetricsWsClientModule.HttpClientConnParams.HTTP_TIMEOUT, METRICS_WS_HTTP_TIMEOUT);
079    return metricsProperties;
080  }
081
082  public RegistryWsServletListener() throws IOException {
083    super(PropertiesUtil.readFromFile(ConfUtils.getAppConfFile(APP_CONF_FILE)), PACKAGES, true, responseFilters, requestFilters);
084  }
085
086  @VisibleForTesting
087  public RegistryWsServletListener(Properties properties) {
088    super(properties, PACKAGES, true, null, requestFilters);
089  }
090
091  @Override
092  protected List<Module> getModules(Properties properties) {
093    return Lists.newArrayList(new DoiModule(properties),
094                              new RegistryMyBatisModule(properties),
095                              new DrupalMyBatisModule(properties),
096                              new ImsModule(),
097                              StringTrimInterceptor.newMethodInterceptingModule(),
098                              new ValidationModule(),
099                              new EventModule(properties),
100                              new RegistrySearchModule(properties),
101                              new SecurityModule(properties),
102                              new VarnishPurgeModule(properties),
103                              new TitleLookupModule(true, properties.getProperty(API_URL_PROPERTY)),
104                              new MetricsWsClientModule(getMetricsProperties(properties)),
105                              new OaipmhModule(properties));
106  }
107
108  @VisibleForTesting
109  @Override
110  protected Injector getInjector() {
111    return super.getInjector();
112  }
113}
