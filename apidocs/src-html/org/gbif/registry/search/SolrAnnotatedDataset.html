001package org.gbif.registry.search;
002
003import org.gbif.api.model.registry.search.DatasetSearchResult;
004import org.gbif.api.vocabulary.Country;
005import org.gbif.api.vocabulary.DatasetSubtype;
006import org.gbif.api.vocabulary.DatasetType;
007import org.gbif.common.search.model.FacetField;
008import org.gbif.common.search.model.FacetField.Method;
009import org.gbif.common.search.model.FullTextSearchField;
010import org.gbif.common.search.model.Key;
011import org.gbif.common.search.model.SearchMapping;
012import org.gbif.common.search.model.WildcardPadding;
013
014import java.util.List;
015import java.util.Set;
016import java.util.UUID;
017
018import com.google.common.collect.Sets;
019import org.apache.solr.client.solrj.beans.Field;
020
021/**
022 * Annotated version of the dataset search result providing the required mapping to the SOLR schema and translation
023 * between String to UUID as SOLR does not support UUID well.
024 */
025@SearchMapping(
026  facets = {
027    @FacetField(name = "TYPE", field = "dataset_type", sort = FacetField.SortOrder.INDEX, method = Method.ENUM),
028    @FacetField(name = "SUBTYPE", field = "dataset_subtype", sort = FacetField.SortOrder.INDEX, method = Method.ENUM),
029    @FacetField(name = "KEYWORD", field = "keyword", method = Method.ENUM),
030    @FacetField(name = "PUBLISHING_ORG", field = "publishing_organization_key", method = Method.ENUM),
031    @FacetField(name = "HOSTING_ORG", field = "hosting_organization_key", method = Method.ENUM),
032    @FacetField(name = "DECADE", field = "decade", sort = FacetField.SortOrder.INDEX, method = Method.ENUM),
033    @FacetField(name = "COUNTRY", field = "country", sort = FacetField.SortOrder.INDEX, method = Method.ENUM),
034    @FacetField(name = "PUBLISHING_COUNTRY", field = "publishing_country", sort = FacetField.SortOrder.INDEX,
035      method = Method.ENUM)
036  },
037  fulltextFields = {
038    @FullTextSearchField(field = "dataset_title", highlightField = "dataset_title", exactMatchScore = 1000.0d,
039      partialMatchScore = 1.0d),
040    @FullTextSearchField(field = "keyword", partialMatching = WildcardPadding.NONE, exactMatchScore = 4.0d),
041    @FullTextSearchField(field = "publishing_organization_title", highlightField = "publishing_organization_title",
042      partialMatching = WildcardPadding.NONE, exactMatchScore = 2.0d),
043    @FullTextSearchField(field = "hosting_organization_title", partialMatching = WildcardPadding.NONE,
044      exactMatchScore = 2.0d),
045    @FullTextSearchField(field = "description", partialMatching = WildcardPadding.NONE),
046    @FullTextSearchField(field = "metadata", partialMatching = WildcardPadding.NONE, exactMatchScore = 0.5d)
047  })
048public class SolrAnnotatedDataset extends DatasetSearchResult {
049
050  @Field("country")
051  public void setCountryCoverage(List<Integer> countryOrdinals) {
052    Set<Country> countries = Sets.newHashSet();
053    for (Integer ordinal : countryOrdinals) {
054      if (ordinal != null) {
055        countries.add(Country.values()[ordinal]);
056      }
057    }
058    super.setCountryCoverage(countries);
059  }
060
061  @Field("publishing_country")
062  public void setPublishingCountry(Integer countryOrdinal) {
063    super.setPublishingCountry(countryOrdinal == null ? Country.UNKNOWN : Country.values()[countryOrdinal]);
064  }
065
066  @Field("decade")
067  @Override
068  public void setDecades(List<Integer> decades) {
069    super.setDecades(decades);
070  }
071
072  @Field("description")
073  @Override
074  public void setDescription(String description) {
075    super.setDescription(description);
076  }
077
078  @Field("metadata")
079  @Override
080  public void setFullText(String metadata) {
081    super.setFullText(metadata);
082  }
083
084  @Field("hosting_organization_key")
085  public void setHostingOrganizationKey(String hostingOrganizationKey) {
086    setHostingOrganizationKey(UUID.fromString(hostingOrganizationKey));
087  }
088
089  @Override
090  @Field("hosting_organization_title")
091  public void setHostingOrganizationTitle(String hostingOrganizationTitle) {
092    super.setHostingOrganizationTitle(hostingOrganizationTitle);
093  }
094
095  @Field("key")
096  @Key
097  public void setKey(String key) {
098    setKey(UUID.fromString(key));
099  }
100
101  @Override
102  @Field("keyword")
103  public void setKeywords(List<String> keywords) {
104    super.setKeywords(keywords);
105  }
106
107  @Field("publishing_organization_key")
108  public void setPublishingOrganizationKey(String publishingOrganizationKey) {
109    setPublishingOrganizationKey(UUID.fromString(publishingOrganizationKey));
110  }
111
112  @Override
113  @Field("publishing_organization_title")
114  public void setPublishingOrganizationTitle(String publishingOrganizationTitle) {
115    super.setPublishingOrganizationTitle(publishingOrganizationTitle);
116  }
117
118  @Field("dataset_subtype")
119  public void setSubtype(Integer datasetSubtypeOrdinal) {
120    setSubtype(datasetSubtypeOrdinal == null ? null : DatasetSubtype.values()[datasetSubtypeOrdinal]);
121  }
122
123  @Field("dataset_title")
124  @Override
125  public void setTitle(String title) {
126    super.setTitle(title);
127  }
128
129  @Field("dataset_type")
130  public void setType(Integer datasetTypeOrdinal) {
131    setType(datasetTypeOrdinal == null ? null : DatasetType.values()[datasetTypeOrdinal]);
132  }
133}
