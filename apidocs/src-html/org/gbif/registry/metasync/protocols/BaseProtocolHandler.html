046import static org.apache.commons.digester3.binder.DigesterLoader.newLoader;

048/**
049 * This class is used as a base for all protocols and provides common methods to be used by all protocols.
050 */
051public abstract class BaseProtocolHandler implements MetadataProtocolHandler {
052
053  private static final Logger LOG = LoggerFactory.getLogger(BaseProtocolHandler.class);
054  private final HttpClient httpClient;
055
056  protected BaseProtocolHandler(HttpClient httpClient) {
057    this.httpClient = httpClient;
058  }
059
060  /**
061   * Takes two collections of {@link Contact}s and tries to match the {@code newContacts} to the
062   * {@code existingContacts} and returns all Contacts that could <b>not</b> be matched (i.e. those are new Contacts
063   * that did
064   * not previously exist).
065   * 
066   * @return new Contacts that do not exist in the existing Contacts already
067   */
068  protected List<Contact> matchContacts(Iterable<Contact> existingContacts, Iterable<Contact> newContacts) {
069    List<Contact> resultList = Lists.newArrayList();
070    for (Contact newContact : newContacts) {
071      boolean found = false;
072      for (Contact existingContact : existingContacts) {
073        if (existingContact.lenientEquals(newContact)) {
074          resultList.add(existingContact);
075          found = true;
076          break;
077        }
078      }
079
080      if (!found) {
081        resultList.add(newContact);
082      }
083    }
084
085    return resultList;
086  }
087
088  /**
089   * Makes a HTTP request to the provided {@link URI} and uses the {@link Digester} to parse the response.
090   * 
091   * @param uri to issue request against
092   * @param digester to parse response with
093   * @param <T> type of Object to return
094   * @throws MetadataException in case anything goes wrong during the request, all underlying HTTP errors are mapped to
095   *         the appropriate {@link ErrorCode}s.
096   */
097  protected <T> T doHttpRequest(URI uri, Digester digester) throws MetadataException {
098    LOG.info("Issuing request: {}", uri);
099    HttpUriRequest get = new HttpGet(uri);
100    HttpResponse response;
101    try {
102      // Not using a Response Handler here because that can't throw arbitrary Exceptions
103      response = httpClient.execute(get);
104    } catch (ClientProtocolException e) {
105      throw new MetadataException(e, ErrorCode.HTTP_ERROR);
106    } catch (IOException e) {
107      throw new MetadataException(e, ErrorCode.IO_EXCEPTION);
108    }
109
110    // Everything but HTTP status 200 is an error
111    if (response.getStatusLine().getStatusCode() != 200) {
112      LOG.debug("Received HTTP code[{}] cause[{}] for request: {}", response.getStatusLine().getStatusCode(),
113        response.getStatusLine().getReasonPhrase(), uri);
114      String cause = String.format("Received HTTP code[%d], phrase[%s]", response.getStatusLine().getStatusCode(),
115        response.getStatusLine().getReasonPhrase());
116      throw new MetadataException(cause, ErrorCode.HTTP_ERROR);
117    }
118
119    try {
120      return digester.parse(response.getEntity().getContent());
121    } catch (SAXException e) {
122      throw new MetadataException(e, ErrorCode.PROTOCOL_ERROR);
123    } catch (IOException e) {
124      throw new MetadataException(e, ErrorCode.IO_EXCEPTION);
125    } finally {
126      try {
127        EntityUtils.consume(response.getEntity());
128      } catch (IOException e2) {
129        LOG.warn("Error consuming content after an exception", e2);
130      }
131    }
132
133  }
134
135  /**
136   * Returns a new Digester which is configured with the annotation rules from the passed in class.
137   */
138  protected Digester newDigester(final Class<?> clazz) {
139    DigesterLoader loader = newLoader(new FromAnnotationsRuleModule() {
140
141      @Override
142      protected void configureRules() {
143        bindRulesFrom(clazz);
144      }
145    });
146
147    loader.setNamespaceAware(true);
148    ConvertUtils.register(new DateTimeConverter(), DateTime.class);
149    ConvertUtils.register(new LanguageConverter(), Language.class);
150    ConvertUtils.register(new PeriodConverter(), Period.class);
151    ConvertUtils.register(new UriConverter(), URI.class);
152    return loader.newDigester();
153  }
