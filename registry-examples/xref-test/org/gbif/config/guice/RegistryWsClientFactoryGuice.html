1   package org.gbif.config.guice;
2   
3   import org.gbif.registry.ws.client.guice.RegistryWsClientModule;
4   import org.gbif.ws.client.guice.AnonymousAuthModule;
5   import org.gbif.ws.client.guice.SingleUserAuthModule;
6   
7   import java.util.Properties;
8   
9   import com.google.inject.Guice;
10  import com.google.inject.Injector;
11  
12  public class RegistryWsClientFactoryGuice {
13  
14    // cache everything, for reuse in repeated calls
15    private static Injector webserviceClient;
16    private static Injector webserviceClientReadOnly;
17    private static final Properties properties = new Properties();
18    static {
19      properties.setProperty("registry.ws.url", "http://api.gbif-uat.org/v1/");
20    }
21  
22    // GBIF account
23    private static final String USERNAME = "ws_client_demo";
24    private static final String PASSWORD = "Demo123";
25  
26    /**
27     * @return An injector that is bound for the webservice client layer.
28     */
29    public static synchronized Injector webserviceClient() {
30      if (webserviceClient == null) {
31        // IMPORTANT: GBIF account must exist and must have correct permissions set
32        SingleUserAuthModule auth = new SingleUserAuthModule(USERNAME, PASSWORD);
33        webserviceClient = Guice.createInjector(new RegistryWsClientModule(properties), auth);
34      }
35      return webserviceClient;
36    }
37  
38    /**
39     * @return An injector that is bound for the webservice client layer but read-only.
40     */
41    public static synchronized Injector webserviceClientReadOnly() {
42      if (webserviceClientReadOnly == null) {
43        // Anonymous authentication module used, webservice client will be read-only
44        webserviceClientReadOnly =
45          Guice.createInjector(new RegistryWsClientModule(properties), new AnonymousAuthModule());
46      }
47      return webserviceClientReadOnly;
48    }
49  }
