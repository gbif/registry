001package org.gbif.registry.metasync.protocols.biocase.model.capabilities;
002
003import org.gbif.registry.metasync.protocols.biocase.model.BiocaseConcept;
004
005import java.net.URI;
006import java.util.List;
007
008import com.google.common.base.Objects;
009import com.google.common.collect.Lists;
010import org.apache.commons.digester3.annotations.rules.ObjectCreate;
011import org.apache.commons.digester3.annotations.rules.SetNext;
012import org.apache.commons.digester3.annotations.rules.SetProperty;
013
014@ObjectCreate(pattern = "response/content/capabilities/SupportedSchemas")
015public class SupportedSchema {
016
017  @SetProperty(pattern = "response/content/capabilities/SupportedSchemas", attributeName = "namespace")
018  private URI namespace;
019  @SetProperty(pattern = "response/content/capabilities/SupportedSchemas", attributeName = "request")
020  private boolean request;
021  @SetProperty(pattern = "response/content/capabilities/SupportedSchemas", attributeName = "response")
022  private boolean response;
023  private List<BiocaseConcept> concepts = Lists.newArrayList();
024
025  public URI getNamespace() {
026    return namespace;
027  }
028
029  public void setNamespace(URI namespace) {
030    this.namespace = namespace;
031  }
032
033  public boolean isRequest() {
034    return request;
035  }
036
037  public void setRequest(boolean request) {
038    this.request = request;
039  }
040
041  public boolean isResponse() {
042    return response;
043  }
044
045  public void setResponse(boolean response) {
046    this.response = response;
047  }
048
049  public List<BiocaseConcept> getConcepts() {
050    return concepts;
051  }
052
053  public void setConcepts(List<BiocaseConcept> concepts) {
054    this.concepts = concepts;
055  }
056
057  @SetNext
058  public void addConcept(BiocaseConcept concept) {
059    concepts.add(concept);
060  }
061
062  @Override
063  public String toString() {
064    return Objects.toStringHelper(this)
065      .add("namespace", namespace)
066      .add("request", request)
067      .add("response", response)
068      .add("concepts", concepts)
069      .toString();
070  }
071
072}
