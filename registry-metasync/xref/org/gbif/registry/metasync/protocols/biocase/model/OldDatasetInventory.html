3   import java.util.List;

5   import com.google.common.base.Objects;
6   import com.google.common.collect.Lists;
7   import org.apache.commons.digester3.annotations.rules.CallMethod;
8   import org.apache.commons.digester3.annotations.rules.CallParam;
9   import org.apache.commons.digester3.annotations.rules.ObjectCreate;
10  
11  /**
12   * This is the inventory retrieved by issuing a {@code scan} request used prior to BioCASe 3.4.
13   *
14   * @see NewDatasetInventory
15   */
16  @ObjectCreate(pattern = "response/content")
17  public class OldDatasetInventory {
18  
19    private List<String> datasets = Lists.newArrayList();
20  
21    public List<String> getDatasets() {
22      return datasets;
23    }
24  
25    public void setDatasets(List<String> datasets) {
26      this.datasets = datasets;
27    }
28  
29    @CallMethod(pattern = "response/content/scan/value")
30    public void addDataset(
31      @CallParam(pattern = "response/content/scan/value") String dataset
32    ) {
33      datasets.add(dataset);
34    }
35  
36    @Override
37    public String toString() {
38      return Objects.toStringHelper(this).add("datasets", datasets).toString();
39    }
40  }
