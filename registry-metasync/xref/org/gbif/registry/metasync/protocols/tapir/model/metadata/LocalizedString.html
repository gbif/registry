1   package org.gbif.registry.metasync.protocols.tapir.model.metadata;
2   
3   import org.gbif.api.vocabulary.Language;
4   
5   import java.util.ArrayList;
6   import java.util.List;
7   import java.util.Map;
8   
9   import com.google.common.collect.Maps;
10  
11  public class LocalizedString {
12  
13    private final Map<Language, String> values = Maps.newHashMap();
14  
15    public void addValue(Language language, String value) {
16      values.put(language, value);
17    }
18  
19    public Map<Language, String> getValues() {
20      return values;
21    }
22  
23    @Override
24    public String toString() {
25      if (values.isEmpty()) {
26        return "";
27      }
28  
29      if (values.containsKey(Language.ENGLISH)) {
30        return values.get(Language.ENGLISH);
31      }
32  
33      List<Language> languages = new ArrayList<Language>(values.keySet());
34      return values.get(languages.get(0));
35    }
36  }
