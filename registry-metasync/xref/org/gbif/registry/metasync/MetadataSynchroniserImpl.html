34  /**
35   * A concrete implementation of the {@link MetadataSynchroniser} interface. It delegates the actual interaction with
36   * Installations and Endpoints to {@link MetadataProtocolHandler}s which have to be registered on this class using
37   * {@link #registerProtocolHandler(MetadataProtocolHandler)}.
38   */
39  public class MetadataSynchroniserImpl implements MetadataSynchroniser {
40  
41    private static final Logger LOG = LoggerFactory.getLogger(MetadataSynchroniserImpl.class);
42    private static final int PAGING_LIMIT = 10000;
43    private final InstallationService installationService;
44    private final List<MetadataProtocolHandler> protocolHandlers = Lists.newArrayList();
45  
46    @Inject
47    public MetadataSynchroniserImpl(InstallationService installationService) {
48      this.installationService = installationService;
49    }
50  
51    @Override
52    public SyncResult synchroniseInstallation(UUID key) {
53      checkNotNull(key, "key can't be null");
54  
55      Installation installation = validateInstallation(key);
56      List<Dataset> hostedDatasets = getHostedDatasets(key);
57  
58      for (MetadataProtocolHandler protocolHandler : protocolHandlers) {
59        if (protocolHandler.canHandle(installation)) {
60          return doSynchroniseInstallation(installation, hostedDatasets, protocolHandler);
61        }
62      }
63  
64      throw new IllegalArgumentException("Installation of type [" + installation.getType() + "] not supported");
65    }
66  
67    @Override
68    public List<SyncResult> synchroniseAllInstallations() {
69      return synchroniseAllInstallations(1);
70    }
71  
72    @Override
73    public List<SyncResult> synchroniseAllInstallations(int parallel) {
74      checkArgument(parallel > 0, "parallel has to be greater than 0");
75      ExecutorService executor = Executors.newFixedThreadPool(parallel);
76      CompletionService<SyncResult> completionService = new ExecutorCompletionService<SyncResult>(executor);
77  
78      // Submit all sync jobs to the Completion service
79      PagingResponse<Installation> results;
80      PagingRequest page = new PagingRequest();
81      int count = 0;
82      do {
83        results = installationService.list(page);
84        // TODO:Check for null result
85        for (final Installation installation : results.getResults()) {
86          completionService.submit(new Callable<SyncResult>() {
87  
88            @Override
89            public SyncResult call() throws Exception {
90              try {
91                return synchroniseInstallation(installation.getKey());
92              } catch (IllegalArgumentException e) {
93                return new SyncResult(installation, new MetadataException(e, ErrorCode.OTHER_ERROR));
94              }
95            }
96          });
97          count++;
98        }
99        page.nextPage();
100     } while (!results.isEndOfRecords());
101 
102     // Wait for all to finish and collect results
103     List<SyncResult> syncResults = Lists.newArrayList();
104     for (int i = 0; i < count; i++) {
105       try {
106         Future<SyncResult> future = completionService.take();
107         syncResults.add(future.get());
108       } catch (InterruptedException ignored) {
109         Thread.currentThread().interrupt();
110       } catch (ExecutionException e) {
111         Throwables.propagateIfPossible(e.getCause());
112       }
113     }
114 
115     executor.shutdown();
116     while (!executor.isTerminated()) {
117       LOG.info("Waiting for synchronisations to finish");
118       try {
119         executor.awaitTermination(1, TimeUnit.MINUTES);
120       } catch (InterruptedException ignored) {
121         Thread.currentThread().interrupt();
122       }
123     }
124 
125     return syncResults;
126   }
127 
128   public void registerProtocolHandler(MetadataProtocolHandler handler) {
129     protocolHandlers.add(handler);
130   }
131 
132   /**
133    * This method actually runs the metadata synchronisation by calling out to the protocol handler and then validating
134    * and processing its result.
135    */
136   private SyncResult doSynchroniseInstallation(
137     Installation installation, List<Dataset> hostedDatasets, MetadataProtocolHandler protocolHandler
138     ) {
139     LOG.info("Syncing Installation [{}] of type [{}]", installation.getKey(), installation.getType());
140     try {
141       return protocolHandler.syncInstallation(installation, hostedDatasets);
142     } catch (MetadataException e) {
143       return new SyncResult(installation, e);
144     }
145   }
146 
147   /**
148    * Does some checks whether we can synchronise this Installation or not. They are not exhaustive as some things can
149    * only be determined by the protocol handlers.
150    */
151   private Installation validateInstallation(UUID key) {
152     Installation installation = installationService.get(key);
153 
154     if (installation == null) {
155       throw new IllegalArgumentException("Installation with key [" + key + "] does not exist");
156     }
157 
158     if (installation.getDeleted() != null) {
159       throw new IllegalArgumentException("Installation with key [" + key +"] is deleted");
160     }
161 
162     if (installation.getEndpoints() == null || installation.getEndpoints().isEmpty()) {
163       throw new IllegalArgumentException("Installation with key [" + key + "]" + " has no endpoints");
164     }
165     return installation;
166   }
167 
168   /**
169    * Gets all hosted datasets for an Installation.
170    * 
171    * @param key of the Installation
172    * @return list of Datasets for this Installation, might be empty but never null
173    */
174   private List<Dataset> getHostedDatasets(UUID key) {
175     PagingRequest page = new PagingRequest(0, PAGING_LIMIT);
176     PagingResponse<Dataset> results;
177     List<Dataset> hostedDatasets = Lists.newArrayList();
178     do {
179       results = installationService.getHostedDatasets(key, page);
180       hostedDatasets.addAll(results.getResults());
181       page.nextPage();
182     } while (!results.isEndOfRecords());
183     return hostedDatasets;
184   }
