8   import java.util.Collection;
9   import java.util.List;
10  import java.util.Map;
11  
12  import com.google.common.base.Preconditions;
13  
14  /**
15   * A simple holder object used to pass around the result of metadata synchronisation.
16   */
17  @SuppressWarnings({"PublicField", "AssignmentToCollectionOrArrayFieldFromParameter"})
18  public class SyncResult {
19  
20    /**
21     * Maps from the existing Dataset in the Registry to a new object that we just parsed from the Endpoint.
22     */
23    public Map<Dataset, Dataset> existingDatasets;
24    public List<Dataset> addedDatasets;
25    public List<Dataset> deletedDatasets;
26    public Installation installation;
27    public MetadataException exception;
28  
29    public SyncResult(
30      Map<Dataset, Dataset> existingDatasets,
31      List<Dataset> addedDatasets,
32      List<Dataset> deletedDatasets,
33      Installation installation
34    ) {
35      this.existingDatasets = existingDatasets;
36      this.addedDatasets = addedDatasets;
37      this.deletedDatasets = deletedDatasets;
38      this.installation = installation;
39    }
40  
41    public SyncResult(Installation installation, MetadataException exception) {
42      this.installation = installation;
43      this.exception = exception;
44    }
45  
46    /**
47     * @return A metasync history summary of the result
48     */
49    public MetasyncHistory buildHistory() {
50      MetasyncHistory history = new MetasyncHistory();
51      history.setInstallationKey(installation.getKey());
52  
53      if (exception != null) {
54        history.setResult(buildCode(exception.getErrorCode()));
55        String message = String.format(
56          "Synchronization failed with error [%s]. %d datasets were updated. %d datasets were added. %d datasets were deleted.",
57          exception.getMessage(),
58          sizeOf(existingDatasets),
59          sizeOf(addedDatasets),
60          sizeOf(deletedDatasets));
61        history.setDetails(message);
62      } else {
63        history.setResult(MetasyncResult.OK);
64        String message = String.format(
65          "Synchronization succeeded. %d datasets were updated. %d datasets were added. %d datasets were deleted.",
66          sizeOf(existingDatasets),
67          sizeOf(addedDatasets),
68          sizeOf(deletedDatasets));
69        history.setDetails(message);
70      }
71      return history;
72    }
73  
74    // NPE safe version
75    private int sizeOf(Collection<?> collection) {
76      return collection == null ? 0 : collection.size();
77    }
78  
79    // NPE safe version
80    private int sizeOf(Map<?, ?> collection) {
81      return collection == null ? 0 : collection.size();
82    }
83  
84    private MetasyncResult buildCode(ErrorCode code) {
85      Preconditions.checkNotNull(code, "Cannot build a code from null input");
86      switch (code) {
87        case IO_EXCEPTION:
88          return MetasyncResult.IO_EXCEPTION;
89        case PROTOCOL_ERROR:
90          return MetasyncResult.PROTOCOL_ERROR;
91        case HTTP_ERROR:
92          return MetasyncResult.HTTP_ERROR;
93        case OTHER_ERROR:
94          return MetasyncResult.OTHER_ERROR;
95        default:
96          return MetasyncResult.OTHER_ERROR;
97      }
98    }
99  }
