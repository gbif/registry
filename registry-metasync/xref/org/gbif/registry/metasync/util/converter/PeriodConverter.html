1   package org.gbif.registry.metasync.util.converter;
2   
3   import org.apache.commons.beanutils.Converter;
4   import org.joda.time.Period;
5   import org.joda.time.format.ISOPeriodFormat;
6   import org.joda.time.format.PeriodFormatter;
7   
8   import static com.google.common.base.Preconditions.checkArgument;
9   import static com.google.common.base.Preconditions.checkNotNull;
10  
11  /**
12   * Used by commons-digester (via commons-beanutils) to convert Strings into {@link Period}s from Joda Time.
13   */
14  public class PeriodConverter implements Converter {
15  
16    private final PeriodFormatter formatter = ISOPeriodFormat.standard();
17  
18    @Override
19    public Object convert(Class type, Object value) {
20      checkNotNull(type, "type cannot be null");
21      checkNotNull(value, "Value cannot be null");
22      checkArgument(type.equals(Period.class),
23                    "Conversion target should be org.joda.time.Duration, but is %s",
24                    type.getClass());
25      checkArgument(String.class.isAssignableFrom(value.getClass()),
26                    "Value should be a string, but is a %s",
27                    value.getClass());
28  
29      return formatter.parsePeriod((String) value);
30    }
31  
32  }
