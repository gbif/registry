1   package org.gbif.registry.metasync.resulthandler;
2   
3   import org.gbif.registry.metasync.api.SyncResult;
4   
5   import org.slf4j.Logger;
6   import org.slf4j.LoggerFactory;
7   
8   /**
9    * A simple utility class that prints some information about each synchronisation result. Does not generate aggregate
10   * statistics at the moment.
11   */
12  public final class DebugHandler {
13  
14    private static final Logger LOG = LoggerFactory.getLogger(DebugHandler.class);
15  
16    public static void processResults(Iterable<SyncResult> results) {
17      for (SyncResult result : results) {
18        processResult(result);
19      }
20    }
21  
22    public static void processResult(SyncResult result) {
23      if (result == null) {
24        LOG.warn("Null SyncResult!");
25      } else if (result.exception == null) {
26        LOG.info("Installation [{}] synced successfully. [{}] added, [{}] deleted, [{}] updated",
27                 result.installation.getKey(),
28                 result.addedDatasets.size(),
29                 result.deletedDatasets.size(),
30                 result.existingDatasets.size());
31      } else {
32        LOG.info("Installation [{}] failed sync. Reason: [{}]",
33                 result.installation.getKey(),
34                 result.exception.getErrorCode());
35        if (result.exception.getMessage() != null) {
36          LOG.info("Message: [{}]", result.exception.getMessage());
37        }
38        if (result.exception.getCause() != null) {
39          LOG.info("Cause: [{}], [{}]",
40                   result.exception.getCause().getClass().toString(),
41                   result.exception.getCause().getMessage());
42        }
43      }
44    }
45  
46    private DebugHandler() {
47      throw new UnsupportedOperationException("Can't initialize class");
48    }
49  
50  }
