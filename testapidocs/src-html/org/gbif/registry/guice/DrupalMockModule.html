001package org.gbif.registry.guice;
002
003import org.gbif.api.model.registry.Contact;
004import org.gbif.api.model.registry.Node;
005import org.gbif.api.service.common.UserService;
006import org.gbif.api.vocabulary.ContactType;
007import org.gbif.drupal.mybatis.ImsNodeMapper;
008
009import com.google.inject.AbstractModule;
010
011/**
012 * Mocks the regular DrupalMyBatisModule.
013 */
014public class DrupalMockModule extends AbstractModule {
015
016    @Override
017    protected void configure() {
018      bind(UserService.class).to(UserServiceMock.class);
019      bind(ImsNodeMapper.class).to(ImsNodeMapperMock.class);
020    }
021
022  private static class ImsNodeMapperMock implements ImsNodeMapper {
023    @Override
024    public Node get(Integer integer) {
025      Node mock = new Node();
026      mock.setParticipantSince(2001);
027      mock.setAbbreviation("GBIF.ES");
028      mock.setCity("Madrid");
029      mock.setPostalCode("E-28014");
030      mock.setOrganization("Real Jardín Botánico - CSIC");
031      for (int i = 1; i<8; i++) {
032        Contact c = new Contact();
033        c.setKey(i);
034        c.setLastName("Name " + i);
035        c.setType(ContactType.NODE_STAFF);
036        mock.getContacts().add(c);
037      }
038      return mock;
039    }
040  }
041}
