001package org.gbif.registry.metadata.parse.converter;
002
003import org.gbif.api.vocabulary.Language;
004
005import org.junit.Test;
006
007import static org.junit.Assert.assertEquals;
008import static org.junit.Assert.assertNull;
009
010public class LanguageTypeConverterTest {
011  private LanguageTypeConverter converter = new LanguageTypeConverter();
012
013  @Test
014  public void parseLanguages() throws Throwable {
015    assertLang(Language.AFRIKAANS, "af");
016    assertLang(Language.AFRIKAANS, "AF");
017    assertLang(Language.ENGLISH, "EN");
018    assertLang(Language.ENGLISH, "en_US");
019    assertLang(Language.ENGLISH, "en_UK");
020    assertLang(Language.FRENCH, "french");
021    assertLang(Language.FRENCH, "fra");
022    assertLang(Language.FRENCH, "fre");
023  }
024
025  private void assertLang(Language expect, String value) throws Throwable {
026    Language l = (Language) converter.convertToType(Language.class, value);
027    if (expect == null) {
028      assertNull(l);
029    } else {
030      assertEquals(expect, l);
031    }
032  }
033}
