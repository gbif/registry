001package org.gbif.registry.metadata;
002
003import static org.junit.Assert.fail;
004
005import java.io.IOException;
006import java.io.StringReader;
007import javax.xml.transform.stream.StreamSource;
008import javax.xml.validation.Schema;
009import javax.xml.validation.SchemaFactory;
010import javax.xml.validation.Validator;
011
012import org.xml.sax.SAXException;
013
014/**
015 * XML validation helper class.
016 *
017 * @author cgendreau
018 */
019public class XMLValidator {
020
021  private static final SchemaFactory FACTORY = SchemaFactory.newInstance("http://www.w3.org/2001/XMLSchema");
022
023  /**
024   * Asserts that the xmlContent is valid according to the XSD at xsdLocation.
025   *
026   * @param xmlContent
027   * @param xsdLocation
028   * @throws IOException
029   */
030  public static void assertXMLAgainstXSD(String xmlContent, String xsdLocation) throws IOException {
031    try {
032      //warning, this is slow
033      Schema schema = FACTORY.newSchema(new StreamSource(xsdLocation));
034      Validator validator = schema.newValidator();
035      validator.validate(new StreamSource(new StringReader(xmlContent)));
036    } catch (SAXException saxEx) {
037      fail("XML content does not validate against XSD Schema: " + saxEx.getMessage());
038    }
039  }
040}
