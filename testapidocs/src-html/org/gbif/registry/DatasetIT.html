251  @Test
252  public void testSearchListener() {
253    Dataset d = newEntity();
254    d = create(d, 1);
255    assertSearch(d.getTitle(), 1); // 1 result expected
256
257    // update
258    String oldTitle = d.getTitle();
259    d.setTitle("NEW-DATASET-TITLE");
260    service.update(d);
261    assertSearch("*", 1);
262    assertSearch(oldTitle, 0);
263    assertSearch(d.getTitle(), 1);
264
265    // update publishing organization title should be captured
266    Organization publisher = organizationService.get(d.getPublishingOrganizationKey());
267    assertSearch(publisher.getTitle(), 1);
268    oldTitle = publisher.getTitle();
269    publisher.setTitle("NEW-OWNER-TITLE");
270    organizationService.update(publisher);
271    assertSearch(oldTitle, 0);
272    assertSearch(publisher.getTitle(), 1);
273
274    // update hosting organization title should be captured
275    Installation installation = installationService.get(d.getInstallationKey());
276    assertNotNull("Installation should be present", installation);
277    Organization host = organizationService.get(installation.getOrganizationKey());
278    assertSearch(host.getTitle(), 1);
279    oldTitle = host.getTitle();
280    host.setTitle("NEW-HOST-TITLE");
281    organizationService.update(host);
282    assertSearch(oldTitle, 0);
283    assertSearch(host.getTitle(), 1);
284
285    // check a deletion removes the dataset for search
286    service.delete(d.getKey());
287    assertSearch(host.getTitle(), 0);
288  }

311   * Utility to verify that after waiting for SOLR to update, the given query returns the expected count of results.
312   */
313  private void assertSearch(String query, int expected) {
314    DatasetSearchUpdateUtils.awaitUpdates(datasetIndexUpdater); // SOLR updates are asynchronous
315    DatasetSearchRequest req = new DatasetSearchRequest();
316    req.setQ(query);
317    SearchResponse<DatasetSearchResult, DatasetSearchParameter> resp = searchService.search(req);
318    assertNotNull(resp.getCount());
319    assertEquals("SOLR does not have the expected number of results for query[" + query + "]", Long.valueOf(expected),
320      resp.getCount());
321  }

324   * Utility to verify that after waiting for SOLR to update, the given query returns the expected count of results.
325   */
326  private void assertSearch(Country publishingCountry, Country country, int expected) {
327    DatasetSearchUpdateUtils.awaitUpdates(datasetIndexUpdater); // SOLR updates are asynchronous

388  public void testDoiChanges() {
389    final DOI external1 = new DOI("10.9999/nonGbif");
390    final DOI external2 = new DOI("10.9999/nonGbif2");
391    // we use the test prefix in tests for GBIF DOIs, see registry-test.properties
392    final DOI gbif2 = new DOI("10.5072/sthelse");
393
394    Dataset src = newEntity();
395    src.setDoi(external1);
396    final UUID key = create(src, 1).getKey();
397    Dataset dataset = service.get(key);
398    assertEquals(external1, dataset.getDoi());
399    assertEquals(0, service.listIdentifiers(key).size());
400
401    dataset.setDoi(null);
402    service.update(dataset);
403    dataset = service.get(key);
404    assertNotNull("DOI should never be null", dataset.getDoi());
405    assertFalse(dataset.getDoi().equals(external1));
406    final DOI originalGBIF = dataset.getDoi();
407    assertThat(service.listIdentifiers(key))
408      .hasSize(1)
409      .extracting("identifier").contains(external1.toString());
410
411    dataset.setDoi(external1);
412    service.update(dataset);
413    dataset = service.get(key);
414    assertEquals(external1, dataset.getDoi());
415    assertThat(service.listIdentifiers(key))
416      .hasSize(1)
417      .extracting("identifier").contains(originalGBIF.toString());
418
419    dataset.setDoi(external2);
420    service.update(dataset);
421    dataset = service.get(key);
422    assertEquals(external2, dataset.getDoi());
423    assertThat(service.listIdentifiers(key))
424      .hasSize(2)
425      .extracting("identifier").contains(originalGBIF.toString(), external1.toString());
426
427    dataset.setDoi(null);
428    service.update(dataset);
429    dataset = service.get(key);
430    assertEquals(originalGBIF, dataset.getDoi());
431    assertThat(service.listIdentifiers(key))
432      .hasSize(2)
433      .extracting("identifier").contains(external1.toString(), external2.toString());
434
435    dataset.setDoi(gbif2);
436    service.update(dataset);
437    dataset = service.get(key);
438    assertEquals(gbif2, dataset.getDoi());
439    assertThat(service.listIdentifiers(key))
440      .hasSize(3)
441      .extracting("identifier").contains(external1.toString(), external2.toString(), originalGBIF.toString());
442
443    dataset.setDoi(external1);
444    service.update(dataset);
445    dataset = service.get(key);
446    assertEquals(external1, dataset.getDoi());
447    assertThat(service.listIdentifiers(key))
448      .hasSize(3)
449      .extracting("identifier").contains(gbif2.toString(), external2.toString(), originalGBIF.toString());
450
451  }

460    Dataset d1 = create(newEntity(), 1);
461
462    // upload a valid EML doc
463    service.insertMetadata(d1.getKey(), FileUtils.classpathStream("metadata/sample.xml"));
464
465    // verify our dataset has changed
466    Dataset d2 = service.get(d1.getKey());
467    assertNotEquals("Dataset should have changed after metadata was uploaded", d1, d2);
468    assertEquals("Tanzanian Entomological Collection", d2.getTitle());
469    assertEquals("Created data should not change", d1.getCreated(), d2.getCreated());
470    assertTrue("Dataset modification date should change", d1.getModified().before(d2.getModified()));
471
472    // check metadata
473    List<Metadata> metadata = service.listMetadata(d1.getKey(), MetadataType.DC);
474    assertTrue("No Dublin Core uplaoded yet", metadata.isEmpty());
475
476    metadata = service.listMetadata(d1.getKey(), MetadataType.EML);
477    assertEquals("Exactly one EML uploaded", 1, metadata.size());
478    assertEquals("Wrong metadata type", MetadataType.EML, metadata.get(0).getType());
479
480    // upload subsequent DC document which has less priority than the previous EML doc
481    service.insertMetadata(d1.getKey(), FileUtils.classpathStream("metadata/worms_dc.xml"));
482    // verify our dataset has not changed
483    Dataset d3 = service.get(d1.getKey());
484    assertEquals("Tanzanian Entomological Collection", d3.getTitle());
485    assertEquals("Created data should not change", d1.getCreated(), d3.getCreated());
486  }

578  private void createCountryDatasets(DatasetType type, Country publishingCountry, int number, Country... countries) {
579    Dataset d = addCountryCoverage(newEntity(), countries);
580    d.setType(type);
581    service.create(d);
582
583    // assign a controlled country based organization
584    Organization org = organizationService.get(d.getPublishingOrganizationKey());
585    org.setCountry(publishingCountry);
586    organizationService.update(org);
587
588    // create datasets for it
589    for (int x = 1; x < number; x++) {
590      d = addCountryCoverage(newEntity(org.getKey(), d.getInstallationKey()));
591      d.setType(type);
592      service.create(d);
593    }
594  }

611  private Dataset addCountryCoverage(Dataset d, Country... countries) {
612    if (countries != null) {
613      for (Country c : countries) {
614        if (c != null) {
615          d.getCountryCoverage().add(c);
616        }
617      }
618    }
619    return d;
